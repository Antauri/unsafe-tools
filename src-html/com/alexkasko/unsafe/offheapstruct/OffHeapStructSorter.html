025    /**
026     * <p>alexkasko: borrowed from {@code https://android.googlesource.com/platform/libcore/+/android-4.2.2_r1/luni/src/main/java/java/util/DualPivotQuicksort.java}
027     * and adapted to {@link OffHeapStructCollection}.
028     *
029     * <p>This class implements the Dual-Pivot Quicksort algorithm by
030     * Vladimir Yaroslavskiy, Jon Bentley, and Joshua Bloch. The algorithm
031     * offers O(n log(n)) performance on many data sets that cause other
032     * quicksorts to degrade to quadratic performance, and is typically
033     * faster than traditional (one-pivot) Quicksort implementations.

035     * @author Vladimir Yaroslavskiy

049        /**
050         * Sorts collection using multiple keys. Every subsequent keys is used on
051         * collections interval where previous key has equal values
052         *
053         * @param collection off-heap struct collection
054         * @param keys vararg keys list
055         */
056        public static void sort(OffHeapStructCollection collection, OffHeapStructSortKey... keys) {
057            List<OffHeapStructSortKey> keyList = new ArrayList<OffHeapStructSortKey>(keys.length);
058            Collections.addAll(keyList, keys);
059            sort(collection, keyList);
060        }

067         * @param keys keys list

070            sort(collection, keys, 0, collection.size());
071        }

073        /**
074         * Sorts collection using multiple keys. Every subsequent keys is used on
075         * collections interval where previous key has equal values
076         *
077         * @param collection off-heap struct collection
078         * @param keys keys list
079         * @param fromIndex start sort collection index
080         * @param toIndex end sort collection index
081         */
082        public static void sort(OffHeapStructCollection collection, List<OffHeapStructSortKey> keys, long fromIndex, long toIndex) {
083            if (fromIndex < 0 || fromIndex > toIndex || toIndex > collection.size()) {
084                throw new IllegalArgumentException("Illegal input, collection size: [" + collection.size() + "], " +
085                        "fromIndex: [" + fromIndex + "], toIndex: [" + toIndex + "]");
086            }
087            if(fromIndex == toIndex) return; // nothing to sort here
088            multisortInternal(collection, fromIndex, toIndex, keys, 0);
089        }
090    
091        private static void multisortInternal(OffHeapStructCollection collection, long start, long end,
092                                         List<OffHeapStructSortKey> keys, int keyIndex) {
093            if(keyIndex >= keys.size()) return;
094            OffHeapStructSortKey key = keys.get(keyIndex);
095            // sort current key
096            sortByKey(collection, start, end, key);
097            // whether last key reached
098            if(keyIndex >= keys.size() - 1) return;
099            // sort children keys recursively
100            int childKeyIndex = keyIndex + 1;
101            long childStart = start;
102            long cur = getByKey(collection, childStart, key);
103            for(long i = start + 1; i < end; i++) {
104                long el = getByKey(collection, i, key);
105                if(el != cur) {
106                    multisortInternal(collection, childStart, i, keys, childKeyIndex);
107                    childStart = i;
108                    cur = getByKey(collection, childStart, key);
109                }
110            }
111            // sort tail
112            if(childStart < end - 1) multisortInternal(collection, childStart, end, keys, childKeyIndex);
113        }

115        private static void sortByKey(OffHeapStructCollection collection, long start, long end, OffHeapStructSortKey key) {
116            String keyname = key.getClass().getName();
117            if(LongKey.class.getName().equals(keyname)) sortByLongKey(collection, start, end, key.offset());
118            else if(UnsignedLongKey.class.getName().equals(keyname)) sortByUnsignedLongKey(collection, start, end, key.offset());
119            else if(IntKey.class.getName().equals(keyname)) sortByIntKey(collection, start, end, key.offset());
120            else if(UnsignedIntKey.class.getName().equals(keyname)) sortByUnsignedIntKey(collection, start, end, key.offset());
121            else if(ShortKey.class.getName().equals(keyname)) sortByShortKey(collection, start, end, key.offset());
122            else if(UnsignedShortKey.class.getName().equals(keyname)) sortByUnsignedShortKey(collection, start, end, key.offset());
123            else if(ByteKey.class.getName().equals(keyname)) sortByByteKey(collection, start, end, key.offset());
124            else if(UnsignedByteKey.class.getName().equals(keyname)) sortByUnsignedByteKey(collection, start, end, key.offset());
125            else throw new IllegalArgumentException(key.toString());
126        }
127    
128        private static long getByKey(OffHeapStructCollection collection, long index, OffHeapStructSortKey key) {
129            String keyname = key.getClass().getName();
130            if(LongKey.class.getName().equals(keyname)) return collection.getLong(index, key.offset());
131            else if(UnsignedLongKey.class.getName().equals(keyname)) return collection.getLong(index, key.offset());
132            else if(IntKey.class.getName().equals(keyname)) return collection.getInt(index, key.offset());
133            else if(UnsignedIntKey.class.getName().equals(keyname)) return collection.getUnsignedInt(index, key.offset());
134            else if(ShortKey.class.getName().equals(keyname)) return collection.getShort(index, key.offset());
135            else if(UnsignedShortKey.class.getName().equals(keyname)) return collection.getUnsignedShort(index, key.offset());
136            else if(ByteKey.class.getName().equals(keyname)) return collection.getByte(index, key.offset());
137            else if(UnsignedByteKey.class.getName().equals(keyname)) return collection.getUnsignedByte(index, key.offset());
138            else throw new IllegalArgumentException(key.toString());
139        }

024/**
025 * <p>alexkasko: borrowed from {@code https://android.googlesource.com/platform/libcore/+/android-4.2.2_r1/luni/src/main/java/java/util/DualPivotQuicksort.java}
026 * and adapted to {@link OffHeapStructCollection}.
027 *
028 * <p>This class implements the Dual-Pivot Quicksort algorithm by
029 * Vladimir Yaroslavskiy, Jon Bentley, and Joshua Bloch. The algorithm
030 * offers O(n log(n)) performance on many data sets that cause other
031 * quicksorts to degrade to quadratic performance, and is typically
032 * faster than traditional (one-pivot) Quicksort implementations.
033 *
034 * @author Vladimir Yaroslavskiy
035 * @author Jon Bentley
036 * @author Josh Bloch
037 *
038 * @version 2009.11.29 m765.827.12i

050     * collections interval where previous key has equal values
051     *
052     * @param collection off-heap struct collection
053     * @param keys vararg keys list
054     */
055    public static void sort(OffHeapStructCollection collection, OffHeapStructSortKey... keys) {
056        List<OffHeapStructSortKey> keyList = new ArrayList<OffHeapStructSortKey>(keys.length);
057        Collections.addAll(keyList, keys);
058        sort(collection, keyList);
059    }

072    /**
073     * Sorts collection using multiple keys. Every subsequent keys is used on
074     * collections interval where previous key has equal values
075     *
076     * @param collection off-heap struct collection
077     * @param keys keys list
078     * @param fromIndex start sort collection index
079     * @param toIndex end sort collection index
080     */
081    public static void sort(OffHeapStructCollection collection, List<OffHeapStructSortKey> keys, long fromIndex, long toIndex) {
082        if (fromIndex < 0 || fromIndex > toIndex || toIndex > collection.size()) {
083            throw new IllegalArgumentException("Illegal input, collection size: [" + collection.size() + "], " +
084                    "fromIndex: [" + fromIndex + "], toIndex: [" + toIndex + "]");
085        }
086        if(fromIndex == toIndex) return; // nothing to sort here
087        multisortInternal(collection, fromIndex, toIndex, keys, 0);
088    }

148    private static void multisortInternal(OffHeapStructCollection collection, long start, long end,
149                                     List<OffHeapStructSortKey> keys, int keyIndex) {
150        if(keyIndex >= keys.size()) return;
151        OffHeapStructSortKey key = keys.get(keyIndex);
152        // sort current key
153        sortByKey(collection, start, end, key);
154        // whether last key reached
155        if(keyIndex >= keys.size() - 1) return;
156        // sort children keys recursively
157        int childKeyIndex = keyIndex + 1;
158        long childStart = start;
159        long cur = getByKey(collection, childStart, key);
160        for(long i = start + 1; i < end; i++) {
161            long el = getByKey(collection, i, key);
162            if(el != cur) {
163                multisortInternal(collection, childStart, i, keys, childKeyIndex);
164                childStart = i;
165                cur = getByKey(collection, childStart, key);
166            }
167        }
168        // sort tail
169        if(childStart < end - 1) multisortInternal(collection, childStart, end, keys, childKeyIndex);
170    }

172    private static void sortByKey(OffHeapStructCollection col, long start, long end, OffHeapStructSortKey key) {
173        switch (key.typeId()) {
174            case LongKey.ID: sortByLongKey(col, start, end, key.offset()); break;
175            case UnsignedLongKey.ID: sortByUnsignedLongKey(col, start, end, key.offset()); break;
176            case IntKey.ID: sortByIntKey(col, start, end, key.offset()); break;
177            case UnsignedIntKey.ID: sortByUnsignedIntKey(col, start, end, key.offset()); break;
178            case ShortKey.ID: sortByShortKey(col, start, end, key.offset()); break;
179            case UnsignedShortKey.ID: sortByUnsignedShortKey(col, start, end, key.offset()); break;
180            case ByteKey.ID: sortByByteKey(col, start, end, key.offset()); break;
181            case UnsignedByteKey.ID: sortByUnsignedByteKey(col, start, end, key.offset()); break;
182            default: throw new IllegalArgumentException(key.toString());
183        }
184    }
185
186    private static long getByKey(OffHeapStructCollection col, long index, OffHeapStructSortKey key) {
187        switch (key.typeId()) {
188            case LongKey.ID: return col.getLong(index, key.offset());
189            case UnsignedLongKey.ID: return col.getLong(index, key.offset());
190            case IntKey.ID: return col.getInt(index, key.offset());
191            case UnsignedIntKey.ID: return col.getUnsignedInt(index, key.offset());
192            case ShortKey.ID: return col.getShort(index, key.offset());
193            case UnsignedShortKey.ID: return col.getUnsignedShort(index, key.offset());
194            case ByteKey.ID: return col.getByte(index, key.offset());
195            case UnsignedByteKey.ID: return col.getUnsignedByte(index, key.offset());
196            default: throw new IllegalArgumentException(key.toString());
197        }
198    }

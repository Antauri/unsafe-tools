026/**
027 * <p>alexkasko: borrowed from {@code https://android.googlesource.com/platform/libcore/+/android-4.2.2_r1/luni/src/main/java/java/util/DualPivotQuicksort.java}
028 * and adapted to {@link OffHeapStructCollection}.
029 *
030 * <p>This class implements the Dual-Pivot Quicksort algorithm by
031 * Vladimir Yaroslavskiy, Jon Bentley, and Joshua Bloch. The algorithm
032 * offers O(n log(n)) performance on many data sets that cause other
033 * quicksorts to degrade to quadratic performance, and is typically
034 * faster than traditional (one-pivot) Quicksort implementations.
035 *
036 * @author Vladimir Yaroslavskiy
037 * @author Jon Bentley
038 * @author Josh Bloch
039 *
040 * @version 2009.11.29 m765.827.12i

051     * Sorts collection using multiple keys. Every subsequent keys is used on
052     * collections interval where previous key has equal values
053     *
054     * @param collection off-heap struct collection
055     * @param keys vararg keys list
056     */
057    public static void sort(OffHeapStructCollection collection, OffHeapStructSortKey... keys) {
058        List<OffHeapStructSortKey> keyList = new ArrayList<OffHeapStructSortKey>(keys.length);
059        Collections.addAll(keyList, keys);
060        sort(collection, keyList);
061    }

074    /**
075     * Sorts collection using multiple keys. Every subsequent keys is used on
076     * collections interval where previous key has equal values
077     *
078     * @param collection off-heap struct collection
079     * @param keys keys list
080     * @param fromIndex start sort collection index
081     * @param toIndex end sort collection index
082     */
083    public static void sort(OffHeapStructCollection collection, List<OffHeapStructSortKey> keys, long fromIndex, long toIndex) {
084        if (fromIndex < 0 || fromIndex > toIndex || toIndex > collection.size()) {
085            throw new IllegalArgumentException("Illegal input, collection size: [" + collection.size() + "], " +
086                    "fromIndex: [" + fromIndex + "], toIndex: [" + toIndex + "]");
087        }
088        if(fromIndex == toIndex) return; // nothing to sort here
089        multisortInternal(collection, fromIndex, toIndex, keys, 0);
090    }

115    private static void multisortInternal(OffHeapStructCollection collection, long start, long end,
116                                     List<OffHeapStructSortKey> keys, int keyIndex) {
117        if(keyIndex >= keys.size()) return;
118        OffHeapStructSortKey key = keys.get(keyIndex);
119        // sort current key
120        sortByKey(collection, start, end, key);
121        // whether last key reached
122        if(keyIndex >= keys.size() - 1) return;
123        // sort children keys recursively
124        int childKeyIndex = keyIndex + 1;
125        long childStart = start;
126        long cur = getByKey(collection, childStart, key);
127        for(long i = start + 1; i < end; i++) {
128            long el = getByKey(collection, i, key);
129            if(el != cur) {
130                multisortInternal(collection, childStart, i, keys, childKeyIndex);
131                childStart = i;
132                cur = getByKey(collection, childStart, key);
133            }
134        }
135        // sort tail
136        if(childStart < end - 1) multisortInternal(collection, childStart, end, keys, childKeyIndex);
137    }

139    private static void sortByKey(OffHeapStructCollection col, long start, long end, OffHeapStructSortKey key) {
140        switch (key.typeId()) {
141            case LongKey.ID: sortByLongKey(col, start, end, key.offset()); break;
142            case UnsignedLongKey.ID: sortByUnsignedLongKey(col, start, end, key.offset()); break;
143            case IntKey.ID: sortByIntKey(col, start, end, key.offset()); break;
144            case UnsignedIntKey.ID: sortByUnsignedIntKey(col, start, end, key.offset()); break;
145            case ShortKey.ID: sortByShortKey(col, start, end, key.offset()); break;
146            case UnsignedShortKey.ID: sortByUnsignedShortKey(col, start, end, key.offset()); break;
147            case ByteKey.ID: sortByByteKey(col, start, end, key.offset()); break;
148            case UnsignedByteKey.ID: sortByUnsignedByteKey(col, start, end, key.offset()); break;
149            default: throw new IllegalArgumentException(key.toString());
150        }
151    }
152
153    private static long getByKey(OffHeapStructCollection col, long index, OffHeapStructSortKey key) {
154        switch (key.typeId()) {
155            case LongKey.ID: return col.getLong(index, key.offset());
156            case UnsignedLongKey.ID: return col.getLong(index, key.offset());
157            case IntKey.ID: return col.getInt(index, key.offset());
158            case UnsignedIntKey.ID: return col.getUnsignedInt(index, key.offset());
159            case ShortKey.ID: return col.getShort(index, key.offset());
160            case UnsignedShortKey.ID: return col.getUnsignedShort(index, key.offset());
161            case ByteKey.ID: return col.getByte(index, key.offset());
162            case UnsignedByteKey.ID: return col.getUnsignedByte(index, key.offset());
163            default: throw new IllegalArgumentException(key.toString());
164        }
165    }

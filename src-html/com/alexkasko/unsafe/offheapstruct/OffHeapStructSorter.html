076         * Sorts collection using multiple keys. Every subsequent keys is used on
077         * collections interval where previous key has equal values
078         *
079         * @param collection off-heap struct collection
080         * @param keys keys list
081         * @param fromIndex start sort collection index
082         * @param toIndex end sort collection index
083         */
084        public static void sort(OffHeapStructCollection collection, List<OffHeapStructSortKey> keys, long fromIndex, long toIndex) {
085            if (fromIndex < 0 || fromIndex > toIndex || toIndex > collection.size()) {
086                throw new IllegalArgumentException("Illegal input, collection size: [" + collection.size() + "], " +
087                        "fromIndex: [" + fromIndex + "], toIndex: [" + toIndex + "]");
088            }
089            if(fromIndex == toIndex) return; // nothing to sort here
090            multisortInternal(collection, fromIndex, toIndex, keys, 0);
091        }

093        private static void multisortInternal(OffHeapStructCollection collection, long start, long end,
094                                         List<OffHeapStructSortKey> keys, int keyIndex) {
095            if(keyIndex >= keys.size()) return;
096            OffHeapStructSortKey key = keys.get(keyIndex);
097            // sort current key
098            sortByKey(collection, start, end, key);
099            // whether last key reached
100            if(keyIndex >= keys.size() - 1) return;
101            // sort children keys recursively
102            int childKeyIndex = keyIndex + 1;
103            long childStart = start;
104            long cur = getByKey(collection, childStart, key);
105            for(long i = start + 1; i < end; i++) {
106                long el = getByKey(collection, i, key);
107                if(el != cur) {
108                    multisortInternal(collection, childStart, i, keys, childKeyIndex);
109                    childStart = i;
110                    cur = getByKey(collection, childStart, key);
111                }
112            }
113            // sort tail
114            if(childStart < end - 1) multisortInternal(collection, childStart, end, keys, childKeyIndex);
115        }

117        private static void sortByKey(OffHeapStructCollection collection, long start, long end, OffHeapStructSortKey key) {
118            switch (key.length()) {
119                case 8: sortByLongKey(collection, start, end, key.offset()); break;
120                case 4: sortByIntKey(collection, start, end, key.offset()); break;
121                case 2: sortByShortKey(collection, start, end, key.offset()); break;
122                case 1: sortByByteKey(collection, start, end, key.offset()); break;
123                default: throw new IllegalArgumentException(key.toString());
124            }
125        }

127        private static long getByKey(OffHeapStructCollection collection, long index, OffHeapStructSortKey key) {
128            switch (key.length()) {
129                case 8: return collection.getLong(index, key.offset());
130                case 4: return collection.getInt(index, key.offset());
131                case 2: return collection.getShort(index, key.offset());
132                case 1: return collection.getByte(index, key.offset());
133                default: throw new IllegalArgumentException(key.toString());
134            }
135        }

174        /**
175         * Sorts the specified range of the off-heap struct collection into ascending order using long struct key.
176         * This method differs from the public {@code sort} method in that the
177         * {@code right} index is inclusive, and it does no range checking on
178         * {@code left} or {@code right}.
179         *
180         * @param a the off-heap header-payload collection to be sorted
181         * @param left the index of the first element, inclusive, to be sorted
182         * @param right the index of the last element, inclusive, to be sorted* @param keyOffset
183         * @param keyOffset long sort key field offset within stuct bounds
184         * @param pi temporary buffer for structs
185         * @param pj temporary buffer for structs
186         * @param pe1 temporary buffer for structs
187         * @param pe2 temporary buffer for structs
188         * @param pe3 temporary buffer for structs
189         * @param pe4 temporary buffer for structs
190         * @param pe5 temporary buffer for structs
191         */
192        private static void doSortByLongKey(OffHeapStructCollection a, long left, long right, int keyOffset, byte[] pi, byte[] pj,
193                                   byte[] pe1, byte[] pe2, byte[] pe3, byte[] pe4, byte[] pe5) {
194            // Use insertion sort on tiny arrays
195            if (right - left + 1 < INSERTION_SORT_THRESHOLD) {
196                for (long i = left + 1; i <= right; i++) {
197                    long ai = a.getLong(i, keyOffset);
198                    a.get(i, pi);
199                    long j;
200                    for (j = i - 1; j >= left && ai < a.getLong(j, keyOffset); j--) {
201                        a.get(j, pj);
202                        a.set(j + 1, pj);
203                    }
204                    a.set(j + 1, pi);
205                }
206            } else { // Use Dual-Pivot Quicksort on large arrays
207                dualPivotQuicksortByLongKey(a, left, right, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
208            }
209        }

211        /**
212         * Sorts the specified range of the off-heap struct collection into ascending order by the
213         * Dual-Pivot Quicksort algorithm using long struct key.
214         *
215         * @param a the off-heap header-payload collection to be sorted
216         * @param left the index of the first element, inclusive, to be sorted
217         * @param right the index of the last element, inclusive, to be sorted
218         * @param keyOffset long sort key field offset within stuct bounds
219         * @param pi temporary buffer for structs
220         * @param pj temporary buffer for structs
221         * @param pe1 temporary buffer for structs
222         * @param pe2 temporary buffer for structs
223         * @param pe3 temporary buffer for structs
224         * @param pe4 temporary buffer for structs
225         * @param pe5 temporary buffer for structs
226         */
227        private static void dualPivotQuicksortByLongKey(OffHeapStructCollection a, long left, long right, int keyOffset, byte[] pi, byte[] pj,
228                                               byte[] pe1, byte[] pe2, byte[] pe3, byte[] pe4, byte[] pe5) {
229            // Compute indices of five evenly spaced elements
230            long sixth = (right - left + 1) / 6;
231            long e1 = left  + sixth;
232            long e5 = right - sixth;
233            long e3 = (left + right) >>> 1; // The midpoint
234            long e4 = e3 + sixth;
235            long e2 = e3 - sixth;
236    
237            // Sort these elements using a 5-element sorting network
238            long ae1 = a.getLong(e1, keyOffset), ae2 = a.getLong(e2, keyOffset), ae3 = a.getLong(e3, keyOffset),
239                    ae4 = a.getLong(e4, keyOffset), ae5 = a.getLong(e5, keyOffset);
240            a.get(e1, pe1); a.get(e2, pe2); a.get(e3, pe3); a.get(e4, pe4); a.get(e5, pe5);
241    
242            if (ae1 > ae2) { long t = ae1; byte[] pt = pe1; ae1 = ae2; pe1 = pe2; ae2 = t; pe2 = pt; }
243            if (ae4 > ae5) { long t = ae4; byte[] pt = pe4; ae4 = ae5; pe4 = pe5; ae5 = t; pe5 = pt; }
244            if (ae1 > ae3) { long t = ae1; byte[] pt = pe1; ae1 = ae3; pe1 = pe3; ae3 = t; pe3 = pt; }
245            if (ae2 > ae3) { long t = ae2; byte[] pt = pe2; ae2 = ae3; pe2 = pe3; ae3 = t; pe3 = pt; }
246            if (ae1 > ae4) { long t = ae1; byte[] pt = pe1; ae1 = ae4; pe1 = pe4; ae4 = t; pe4 = pt; }
247            if (ae3 > ae4) { long t = ae3; byte[] pt = pe3; ae3 = ae4; pe3 = pe4; ae4 = t; pe4 = pt; }
248            if (ae2 > ae5) { long t = ae2; byte[] pt = pe2; ae2 = ae5; pe2 = pe5; ae5 = t; pe5 = pt; }
249            if (ae2 > ae3) { long t = ae2; byte[] pt = pe2; ae2 = ae3; pe2 = pe3; ae3 = t; pe3 = pt; }
250            if (ae4 > ae5) { long t = ae4; byte[] pt = pe4; ae4 = ae5; pe4 = pe5; ae5 = t; pe5 = pt; }
251    
252            a.set(e1, pe1); a.set(e3, pe3); a.set(e5, pe5);
253    
254            /*
255             * Use the second and fourth of the five sorted elements as pivots.
256             * These values are inexpensive approximations of the first and
257             * second terciles of the array. Note that pivot1 <= pivot2.
258             *
259             * The pivots are stored in local variables, and the first and
260             * the last of the elements to be sorted are moved to the locations
261             * formerly occupied by the pivots. When partitioning is complete,
262             * the pivots are swapped back into their final positions, and
263             * excluded from subsequent sorting.
264             */
265            a.get(left, pe1);
266            long pivot1 = ae2; a.set(e2, pe1);
267            a.get(right, pe1);
268            long pivot2 = ae4; a.set(e4, pe1);
269    
270            // Pointers
271            long less  = left  + 1; // The index of first element of center part
272            long great = right - 1; // The index before first element of right part
273    
274            boolean pivotsDiffer = (pivot1 != pivot2);
275    
276            if (pivotsDiffer) {
277                /*
278                 * Partitioning:
279                 *
280                 *   left part         center part                    right part
281                 * +------------------------------------------------------------+
282                 * | < pivot1  |  pivot1 <= && <= pivot2  |    ?    |  > pivot2 |
283                 * +------------------------------------------------------------+
284                 *              ^                          ^       ^
285                 *              |                          |       |
286                 *             less                        k     great
287                 *
288                 * Invariants:
289                 *
290                 *              all in (left, less)   < pivot1
291                 *    pivot1 <= all in [less, k)     <= pivot2
292                 *              all in (great, right) > pivot2
293                 *
294                 * Pointer k is the first index of ?-part
295                 */
296                outer:
297                for (long k = less; k <= great; k++) {
298                    long ak = a.getLong(k, keyOffset);
299                    a.get(k, pe1);
300                    if (ak < pivot1) { // Move a[k] to left part
301                        if (k != less) {
302                            a.get(less, pe3);
303                            a.set(k, pe3);
304                            a.set(less, pe1);
305                        }
306                        less++;
307                    } else if (ak > pivot2) { // Move a[k] to right part
308                        while (a.getLong(great, keyOffset) > pivot2) {
309                            if (great-- == k) {
310                                break outer;
311                            }
312                        }
313                        if (a.getLong(great, keyOffset) < pivot1) {
314                            a.get(less, pe3);
315                            a.set(k, pe3);
316                            a.get(great, pe3);
317                            a.set(less++, pe3);
318                            a.set(great--, pe1);
319                        } else { // pivot1 <= a[great] <= pivot2
320                            a.get(great, pe3);
321                            a.set(k, pe3);
322                            a.set(great--, pe1);
323                        }
324                    }
325                }
326            } else { // Pivots are equal
327                /*
328                 * Partition degenerates to the traditional 3-way,
329                 * or "Dutch National Flag", partition:
330                 *
331                 *   left part   center part            right part
332                 * +----------------------------------------------+
333                 * |  < pivot  |  == pivot  |    ?    |  > pivot  |
334                 * +----------------------------------------------+
335                 *              ^            ^       ^
336                 *              |            |       |
337                 *             less          k     great
338                 *
339                 * Invariants:
340                 *
341                 *   all in (left, less)   < pivot
342                 *   all in [less, k)     == pivot
343                 *   all in (great, right) > pivot
344                 *
345                 * Pointer k is the first index of ?-part
346                 */
347                for (long k = less; k <= great; k++) {
348                    long ak = a.getLong(k, keyOffset);
349                    a.get(k, pe1);
350                    if (ak == pivot1) {
351                        continue;
352                    }
353                    if (ak < pivot1) { // Move a[k] to left part
354                        if (k != less) {
355                            a.get(less, pe3);
356                            a.set(k, pe3);
357                            a.set(less, pe1);
358                        }
359                        less++;
360                    } else { // (a[k] > pivot1) -  Move a[k] to right part
361                        /*
362                         * We know that pivot1 == a[e3] == pivot2. Thus, we know
363                         * that great will still be >= k when the following loop
364                         * terminates, even though we don't test for it explicitly.
365                         * In other words, a[e3] acts as a sentinel for great.
366                         */
367                        while (a.getLong(great, keyOffset) > pivot1) {
368                            great--;
369                        }
370                        if (a.getLong(great, keyOffset) < pivot1) {
371                            a.get(less, pe3);
372                            a.set(k, pe3);
373                            a.get(great, pe3);
374                            a.set(less++, pe3);
375                            a.set(great--, pe1);
376                        } else { // a[great] == pivot1
377                            a.get(great, pe3);
378                            a.set(k, pe3);
379                            a.set(great--, pe1);
380                        }
381                    }
382                }
383            }
384    
385            // Swap pivots into their final positions
386            a.get(less - 1, pe1);
387            a.set(left, pe1); a.set(less - 1, pe2);
388            a.get(great + 1, pe1);
389            a.set(right, pe1); a.set(great + 1, pe4);
390    
391            // Sort left and right parts recursively, excluding known pivot values
392            doSortByLongKey(a, left,   less - 2, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
393            doSortByLongKey(a, great + 2, right, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
394    
395            /*
396             * If pivot1 == pivot2, all elements from center
397             * part are equal and, therefore, already sorted
398             */
399            if (!pivotsDiffer) {
400                return;
401            }
402    
403            /*
404             * If center part is too large (comprises > 2/3 of the array),
405             * swap internal pivot values to ends
406             */
407            if (less < e1 && great > e5) {
408                while (a.getLong(less, keyOffset) == pivot1) {
409                    less++;
410                }
411                while (a.getLong(great, keyOffset) == pivot2) {
412                    great--;
413                }
414    
415                /*
416                 * Partitioning:
417                 *
418                 *   left part       center part                   right part
419                 * +----------------------------------------------------------+
420                 * | == pivot1 |  pivot1 < && < pivot2  |    ?    | == pivot2 |
421                 * +----------------------------------------------------------+
422                 *              ^                        ^       ^
423                 *              |                        |       |
424                 *             less                      k     great
425                 *
426                 * Invariants:
427                 *
428                 *              all in (*, less)  == pivot1
429                 *     pivot1 < all in [less, k)   < pivot2
430                 *              all in (great, *) == pivot2
431                 *
432                 * Pointer k is the first index of ?-part
433                 */
434                outer:
435                for (long k = less; k <= great; k++) {
436                    long ak = a.getLong(k, keyOffset);
437                    a.get(k, pe1);
438                    if (ak == pivot2) { // Move a[k] to right part
439                        while (a.getLong(great, keyOffset) == pivot2) {
440                            if (great-- == k) {
441                                break outer;
442                            }
443                        }
444                        if (a.getLong(great, keyOffset) == pivot1) {
445                            a.get(less, pe3);
446                            a.set(k, pe3);
447                            a.get(great, pe3);
448                            a.set(less++, pe3);
449                        } else { // pivot1 < a[great] < pivot2
450                            a.get(great, pe3);
451                            a.set(k, pe3);
452                        }
453                        a.set(great--, pe1);
454                    } else if (ak == pivot1) { // Move a[k] to left part
455                        a.get(less, pe3);
456                        a.set(k, pe3);
457                        a.set(less++, pe1);
458                    }
459                }
460            }
461    
462            // Sort center part recursively, excluding known pivot values
463            doSortByLongKey(a, less, great, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
464        }

543         *

545         * @param left the index of the first element, inclusive, to be sorted

547         * @param keyOffset int sort key field offset within stuct bounds
548         * @param pi temporary buffer for structs
549         * @param pj temporary buffer for structs
550         * @param pe1 temporary buffer for structs

559            long sixth = (right - left + 1) / 6;
560            long e1 = left  + sixth;
561            long e5 = right - sixth;
562            long e3 = (left + right) >>> 1; // The midpoint
563            long e4 = e3 + sixth;
564            long e2 = e3 - sixth;
565    
566            // Sort these elements using a 5-element sorting network
567            long ae1 = a.getInt(e1, keyOffset), ae2 = a.getInt(e2, keyOffset), ae3 = a.getInt(e3, keyOffset),
568                    ae4 = a.getInt(e4, keyOffset), ae5 = a.getInt(e5, keyOffset);
569            a.get(e1, pe1); a.get(e2, pe2); a.get(e3, pe3); a.get(e4, pe4); a.get(e5, pe5);
570    
571            if (ae1 > ae2) { long t = ae1; byte[] pt = pe1; ae1 = ae2; pe1 = pe2; ae2 = t; pe2 = pt; }
572            if (ae4 > ae5) { long t = ae4; byte[] pt = pe4; ae4 = ae5; pe4 = pe5; ae5 = t; pe5 = pt; }
573            if (ae1 > ae3) { long t = ae1; byte[] pt = pe1; ae1 = ae3; pe1 = pe3; ae3 = t; pe3 = pt; }
574            if (ae2 > ae3) { long t = ae2; byte[] pt = pe2; ae2 = ae3; pe2 = pe3; ae3 = t; pe3 = pt; }
575            if (ae1 > ae4) { long t = ae1; byte[] pt = pe1; ae1 = ae4; pe1 = pe4; ae4 = t; pe4 = pt; }
576            if (ae3 > ae4) { long t = ae3; byte[] pt = pe3; ae3 = ae4; pe3 = pe4; ae4 = t; pe4 = pt; }
577            if (ae2 > ae5) { long t = ae2; byte[] pt = pe2; ae2 = ae5; pe2 = pe5; ae5 = t; pe5 = pt; }
578            if (ae2 > ae3) { long t = ae2; byte[] pt = pe2; ae2 = ae3; pe2 = pe3; ae3 = t; pe3 = pt; }
579            if (ae4 > ae5) { long t = ae4; byte[] pt = pe4; ae4 = ae5; pe4 = pe5; ae5 = t; pe5 = pt; }
580    
581            a.set(e1, pe1); a.set(e3, pe3); a.set(e5, pe5);
582    
583            /*
584             * Use the second and fourth of the five sorted elements as pivots.
585             * These values are inexpensive approximations of the first and
586             * second terciles of the array. Note that pivot1 <= pivot2.
587             *
588             * The pivots are stored in local variables, and the first and
589             * the last of the elements to be sorted are moved to the locations
590             * formerly occupied by the pivots. When partitioning is complete,
591             * the pivots are swapped back into their final positions, and
592             * excluded from subsequent sorting.
593             */
594            a.get(left, pe1);
595            long pivot1 = ae2; a.set(e2, pe1);
596            a.get(right, pe1);
597            long pivot2 = ae4; a.set(e4, pe1);
598    
599            // Pointers
600            long less  = left  + 1; // The index of first element of center part
601            long great = right - 1; // The index before first element of right part
602    
603            boolean pivotsDiffer = (pivot1 != pivot2);
604    
605            if (pivotsDiffer) {
606                /*
607                 * Partitioning:
608                 *
609                 *   left part         center part                    right part
610                 * +------------------------------------------------------------+
611                 * | < pivot1  |  pivot1 <= && <= pivot2  |    ?    |  > pivot2 |
612                 * +------------------------------------------------------------+
613                 *              ^                          ^       ^
614                 *              |                          |       |
615                 *             less                        k     great
616                 *
617                 * Invariants:
618                 *
619                 *              all in (left, less)   < pivot1
620                 *    pivot1 <= all in [less, k)     <= pivot2
621                 *              all in (great, right) > pivot2
622                 *
623                 * Pointer k is the first index of ?-part
624                 */
625                outer:
626                for (long k = less; k <= great; k++) {
627                    long ak = a.getInt(k, keyOffset);
628                    a.get(k, pe1);
629                    if (ak < pivot1) { // Move a[k] to left part
630                        if (k != less) {
631                            a.get(less, pe3);
632                            a.set(k, pe3);
633                            a.set(less, pe1);
634                        }
635                        less++;
636                    } else if (ak > pivot2) { // Move a[k] to right part
637                        while (a.getInt(great, keyOffset) > pivot2) {
638                            if (great-- == k) {
639                                break outer;
640                            }
641                        }
642                        if (a.getInt(great, keyOffset) < pivot1) {
643                            a.get(less, pe3);
644                            a.set(k, pe3);
645                            a.get(great, pe3);
646                            a.set(less++, pe3);
647                            a.set(great--, pe1);
648                        } else { // pivot1 <= a[great] <= pivot2
649                            a.get(great, pe3);
650                            a.set(k, pe3);
651                            a.set(great--, pe1);
652                        }
653                    }
654                }
655            } else { // Pivots are equal
656                /*
657                 * Partition degenerates to the traditional 3-way,
658                 * or "Dutch National Flag", partition:
659                 *
660                 *   left part   center part            right part
661                 * +----------------------------------------------+
662                 * |  < pivot  |  == pivot  |    ?    |  > pivot  |
663                 * +----------------------------------------------+
664                 *              ^            ^       ^
665                 *              |            |       |
666                 *             less          k     great
667                 *
668                 * Invariants:
669                 *
670                 *   all in (left, less)   < pivot
671                 *   all in [less, k)     == pivot
672                 *   all in (great, right) > pivot
673                 *
674                 * Pointer k is the first index of ?-part
675                 */
676                for (long k = less; k <= great; k++) {
677                    long ak = a.getInt(k, keyOffset);
678                    a.get(k, pe1);
679                    if (ak == pivot1) {
680                        continue;
681                    }
682                    if (ak < pivot1) { // Move a[k] to left part
683                        if (k != less) {
684                            a.get(less, pe3);
685                            a.set(k, pe3);
686                            a.set(less, pe1);
687                        }
688                        less++;
689                    } else { // (a[k] > pivot1) -  Move a[k] to right part
690                        /*
691                         * We know that pivot1 == a[e3] == pivot2. Thus, we know
692                         * that great will still be >= k when the following loop
693                         * terminates, even though we don't test for it explicitly.
694                         * In other words, a[e3] acts as a sentinel for great.
695                         */
696                        while (a.getInt(great, keyOffset) > pivot1) {
697                            great--;
698                        }
699                        if (a.getInt(great, keyOffset) < pivot1) {
700                            a.get(less, pe3);
701                            a.set(k, pe3);
702                            a.get(great, pe3);
703                            a.set(less++, pe3);
704                            a.set(great--, pe1);
705                        } else { // a[great] == pivot1
706                            a.get(great, pe3);
707                            a.set(k, pe3);
708                            a.set(great--, pe1);
709                        }
710                    }
711                }
712            }
713    
714            // Swap pivots into their final positions
715            a.get(less - 1, pe1);
716            a.set(left, pe1); a.set(less - 1, pe2);
717            a.get(great + 1, pe1);
718            a.set(right, pe1); a.set(great + 1, pe4);
719    
720            // Sort left and right parts recursively, excluding known pivot values
721            doSortByIntKey(a, left,   less - 2, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
722            doSortByIntKey(a, great + 2, right, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
723    
724            /*
725             * If pivot1 == pivot2, all elements from center
726             * part are equal and, therefore, already sorted
727             */
728            if (!pivotsDiffer) {
729                return;
730            }
731    
732            /*
733             * If center part is too large (comprises > 2/3 of the array),
734             * swap internal pivot values to ends
735             */
736            if (less < e1 && great > e5) {
737                while (a.getInt(less, keyOffset) == pivot1) {
738                    less++;
739                }
740                while (a.getInt(great, keyOffset) == pivot2) {
741                    great--;
742                }
743    
744                /*
745                 * Partitioning:
746                 *
747                 *   left part       center part                   right part
748                 * +----------------------------------------------------------+
749                 * | == pivot1 |  pivot1 < && < pivot2  |    ?    | == pivot2 |
750                 * +----------------------------------------------------------+
751                 *              ^                        ^       ^
752                 *              |                        |       |
753                 *             less                      k     great
754                 *
755                 * Invariants:
756                 *
757                 *              all in (*, less)  == pivot1
758                 *     pivot1 < all in [less, k)   < pivot2
759                 *              all in (great, *) == pivot2
760                 *
761                 * Pointer k is the first index of ?-part
762                 */
763                outer:
764                for (long k = less; k <= great; k++) {
765                    long ak = a.getInt(k, keyOffset);
766                    a.get(k, pe1);
767                    if (ak == pivot2) { // Move a[k] to right part
768                        while (a.getInt(great, keyOffset) == pivot2) {
769                            if (great-- == k) {
770                                break outer;
771                            }
772                        }
773                        if (a.getInt(great, keyOffset) == pivot1) {
774                            a.get(less, pe3);
775                            a.set(k, pe3);
776                            a.get(great, pe3);
777                            a.set(less++, pe3);
778                        } else { // pivot1 < a[great] < pivot2
779                            a.get(great, pe3);
780                            a.set(k, pe3);
781                        }
782                        a.set(great--, pe1);
783                    } else if (ak == pivot1) { // Move a[k] to left part
784                        a.get(less, pe3);
785                        a.set(k, pe3);
786                        a.set(less++, pe1);
787                    }
788                }
789            }
790    
791            // Sort center part recursively, excluding known pivot values
792            doSortByIntKey(a, less, great, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
793        }

941                 * +------------------------------------------------------------+

1094                    long ak = a.getShort(k, keyOffset);
1095                    a.get(k, pe1);
1096                    if (ak == pivot2) { // Move a[k] to right part
1097                        while (a.getShort(great, keyOffset) == pivot2) {
1098                            if (great-- == k) {
1099                                break outer;
1100                            }
1101                        }
1102                        if (a.getShort(great, keyOffset) == pivot1) {
1103                            a.get(less, pe3);
1104                            a.set(k, pe3);
1105                            a.get(great, pe3);
1106                            a.set(less++, pe3);
1107                        } else { // pivot1 < a[great] < pivot2
1108                            a.get(great, pe3);
1109                            a.set(k, pe3);
1110                        }
1111                        a.set(great--, pe1);
1112                    } else if (ak == pivot1) { // Move a[k] to left part
1113                        a.get(less, pe3);
1114                        a.set(k, pe3);
1115                        a.set(less++, pe1);
1116                    }
1117                }
1118            }
1119    
1120            // Sort center part recursively, excluding known pivot values
1121            doSortByShortKey(a, less, great, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
1122        }

1153                        "fromIndex: [" + fromIndex + "], toIndex: [" + toIndex + "]");

1155            int len = a.structLength();
1156            doSortByByteKey(a, fromIndex, toIndex - 1, keyOffset, new byte[len], new byte[len], new byte[len], new byte[len], new byte[len],
1157                    new byte[len], new byte[len]);

1161        /**
1162         * Sorts the specified range of the off-heap struct collection into ascending order using byte struct key.
1163         * This method differs from the public {@code sort} method in that the
1164         * {@code right} index is inclusive, and it does no range checking on
1165         * {@code left} or {@code right}.
1166         *
1167         * @param a the off-heap header-payload collection to be sorted
1168         * @param left the index of the first element, inclusive, to be sorted
1169         * @param right the index of the last element, inclusive, to be sorted* @param keyOffset
1170         * @param keyOffset byte sort key field offset within stuct bounds
1171         * @param pi temporary buffer for structs
1172         * @param pj temporary buffer for structs
1173         * @param pe1 temporary buffer for structs
1174         * @param pe2 temporary buffer for structs
1175         * @param pe3 temporary buffer for structs
1176         * @param pe4 temporary buffer for structs
1177         * @param pe5 temporary buffer for structs
1178         */
1179        private static void doSortByByteKey(OffHeapStructCollection a, long left, long right, int keyOffset, byte[] pi, byte[] pj,
1180                                   byte[] pe1, byte[] pe2, byte[] pe3, byte[] pe4, byte[] pe5) {
1181            // Use insertion sort on tiny arrays
1182            if (right - left + 1 < INSERTION_SORT_THRESHOLD) {
1183                for (long i = left + 1; i <= right; i++) {
1184                    long ai = a.getByte(i, keyOffset);
1185                    a.get(i, pi);
1186                    long j;
1187                    for (j = i - 1; j >= left && ai < a.getByte(j, keyOffset); j--) {
1188                        a.get(j, pj);
1189                        a.set(j + 1, pj);
1190                    }
1191                    a.set(j + 1, pi);
1192                }
1193            } else { // Use Dual-Pivot Quicksort on large arrays
1194                dualPivotQuicksortByByteKey(a, left, right, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
1195            }
1196        }

1198        /**
1199         * Sorts the specified range of the off-heap struct collection into ascending order by the
1200         * Dual-Pivot Quicksort algorithm using byte struct key.
1201         *
1202         * @param a the off-heap header-payload collection to be sorted
1203         * @param left the index of the first element, inclusive, to be sorted
1204         * @param right the index of the last element, inclusive, to be sorted
1205         * @param keyOffset byte sort key field offset within stuct bounds
1206         * @param pi temporary buffer for structs
1207         * @param pj temporary buffer for structs
1208         * @param pe1 temporary buffer for structs
1209         * @param pe2 temporary buffer for structs
1210         * @param pe3 temporary buffer for structs
1211         * @param pe4 temporary buffer for structs
1212         * @param pe5 temporary buffer for structs
1213         */
1214        private static void dualPivotQuicksortByByteKey(OffHeapStructCollection a, long left, long right, int keyOffset, byte[] pi, byte[] pj,
1215                                               byte[] pe1, byte[] pe2, byte[] pe3, byte[] pe4, byte[] pe5) {
1216            // Compute indices of five evenly spaced elements
1217            long sixth = (right - left + 1) / 6;
1218            long e1 = left  + sixth;
1219            long e5 = right - sixth;
1220            long e3 = (left + right) >>> 1; // The midpoint
1221            long e4 = e3 + sixth;
1222            long e2 = e3 - sixth;
1223    
1224            // Sort these elements using a 5-element sorting network
1225            long ae1 = a.getByte(e1, keyOffset), ae2 = a.getByte(e2, keyOffset), ae3 = a.getByte(e3, keyOffset),
1226                    ae4 = a.getByte(e4, keyOffset), ae5 = a.getByte(e5, keyOffset);
1227            a.get(e1, pe1); a.get(e2, pe2); a.get(e3, pe3); a.get(e4, pe4); a.get(e5, pe5);
1228    
1229            if (ae1 > ae2) { long t = ae1; byte[] pt = pe1; ae1 = ae2; pe1 = pe2; ae2 = t; pe2 = pt; }
1230            if (ae4 > ae5) { long t = ae4; byte[] pt = pe4; ae4 = ae5; pe4 = pe5; ae5 = t; pe5 = pt; }
1231            if (ae1 > ae3) { long t = ae1; byte[] pt = pe1; ae1 = ae3; pe1 = pe3; ae3 = t; pe3 = pt; }
1232            if (ae2 > ae3) { long t = ae2; byte[] pt = pe2; ae2 = ae3; pe2 = pe3; ae3 = t; pe3 = pt; }
1233            if (ae1 > ae4) { long t = ae1; byte[] pt = pe1; ae1 = ae4; pe1 = pe4; ae4 = t; pe4 = pt; }
1234            if (ae3 > ae4) { long t = ae3; byte[] pt = pe3; ae3 = ae4; pe3 = pe4; ae4 = t; pe4 = pt; }
1235            if (ae2 > ae5) { long t = ae2; byte[] pt = pe2; ae2 = ae5; pe2 = pe5; ae5 = t; pe5 = pt; }
1236            if (ae2 > ae3) { long t = ae2; byte[] pt = pe2; ae2 = ae3; pe2 = pe3; ae3 = t; pe3 = pt; }
1237            if (ae4 > ae5) { long t = ae4; byte[] pt = pe4; ae4 = ae5; pe4 = pe5; ae5 = t; pe5 = pt; }
1238    
1239            a.set(e1, pe1); a.set(e3, pe3); a.set(e5, pe5);
1240    
1241            /*
1242             * Use the second and fourth of the five sorted elements as pivots.
1243             * These values are inexpensive approximations of the first and
1244             * second terciles of the array. Note that pivot1 <= pivot2.
1245             *
1246             * The pivots are stored in local variables, and the first and
1247             * the last of the elements to be sorted are moved to the locations
1248             * formerly occupied by the pivots. When partitioning is complete,
1249             * the pivots are swapped back into their final positions, and
1250             * excluded from subsequent sorting.
1251             */
1252            a.get(left, pe1);
1253            long pivot1 = ae2; a.set(e2, pe1);
1254            a.get(right, pe1);
1255            long pivot2 = ae4; a.set(e4, pe1);
1256    
1257            // Pointers
1258            long less  = left  + 1; // The index of first element of center part
1259            long great = right - 1; // The index before first element of right part
1260    
1261            boolean pivotsDiffer = (pivot1 != pivot2);
1262    
1263            if (pivotsDiffer) {
1264                /*
1265                 * Partitioning:
1266                 *
1267                 *   left part         center part                    right part
1268                 * +------------------------------------------------------------+
1269                 * | < pivot1  |  pivot1 <= && <= pivot2  |    ?    |  > pivot2 |
1270                 * +------------------------------------------------------------+
1271                 *              ^                          ^       ^
1272                 *              |                          |       |
1273                 *             less                        k     great
1274                 *
1275                 * Invariants:
1276                 *
1277                 *              all in (left, less)   < pivot1
1278                 *    pivot1 <= all in [less, k)     <= pivot2
1279                 *              all in (great, right) > pivot2
1280                 *
1281                 * Pointer k is the first index of ?-part
1282                 */
1283                outer:
1284                for (long k = less; k <= great; k++) {
1285                    long ak = a.getByte(k, keyOffset);
1286                    a.get(k, pe1);
1287                    if (ak < pivot1) { // Move a[k] to left part
1288                        if (k != less) {
1289                            a.get(less, pe3);
1290                            a.set(k, pe3);
1291                            a.set(less, pe1);
1292                        }
1293                        less++;
1294                    } else if (ak > pivot2) { // Move a[k] to right part
1295                        while (a.getByte(great, keyOffset) > pivot2) {
1296                            if (great-- == k) {
1297                                break outer;
1298                            }
1299                        }
1300                        if (a.getByte(great, keyOffset) < pivot1) {
1301                            a.get(less, pe3);
1302                            a.set(k, pe3);
1303                            a.get(great, pe3);
1304                            a.set(less++, pe3);
1305                            a.set(great--, pe1);
1306                        } else { // pivot1 <= a[great] <= pivot2
1307                            a.get(great, pe3);
1308                            a.set(k, pe3);
1309                            a.set(great--, pe1);
1310                        }
1311                    }
1312                }
1313            } else { // Pivots are equal
1314                /*
1315                 * Partition degenerates to the traditional 3-way,
1316                 * or "Dutch National Flag", partition:
1317                 *
1318                 *   left part   center part            right part
1319                 * +----------------------------------------------+
1320                 * |  < pivot  |  == pivot  |    ?    |  > pivot  |
1321                 * +----------------------------------------------+
1322                 *              ^            ^       ^
1323                 *              |            |       |
1324                 *             less          k     great
1325                 *
1326                 * Invariants:
1327                 *
1328                 *   all in (left, less)   < pivot
1329                 *   all in [less, k)     == pivot
1330                 *   all in (great, right) > pivot
1331                 *
1332                 * Pointer k is the first index of ?-part
1333                 */
1334                for (long k = less; k <= great; k++) {
1335                    long ak = a.getByte(k, keyOffset);
1336                    a.get(k, pe1);
1337                    if (ak == pivot1) {
1338                        continue;
1339                    }
1340                    if (ak < pivot1) { // Move a[k] to left part
1341                        if (k != less) {
1342                            a.get(less, pe3);
1343                            a.set(k, pe3);
1344                            a.set(less, pe1);
1345                        }
1346                        less++;
1347                    } else { // (a[k] > pivot1) -  Move a[k] to right part
1348                        /*
1349                         * We know that pivot1 == a[e3] == pivot2. Thus, we know
1350                         * that great will still be >= k when the following loop
1351                         * terminates, even though we don't test for it explicitly.
1352                         * In other words, a[e3] acts as a sentinel for great.
1353                         */
1354                        while (a.getByte(great, keyOffset) > pivot1) {
1355                            great--;
1356                        }
1357                        if (a.getByte(great, keyOffset) < pivot1) {
1358                            a.get(less, pe3);
1359                            a.set(k, pe3);
1360                            a.get(great, pe3);
1361                            a.set(less++, pe3);
1362                            a.set(great--, pe1);
1363                        } else { // a[great] == pivot1
1364                            a.get(great, pe3);
1365                            a.set(k, pe3);
1366                            a.set(great--, pe1);
1367                        }
1368                    }
1369                }
1370            }
1371    
1372            // Swap pivots into their final positions
1373            a.get(less - 1, pe1);
1374            a.set(left, pe1); a.set(less - 1, pe2);
1375            a.get(great + 1, pe1);
1376            a.set(right, pe1); a.set(great + 1, pe4);
1377    
1378            // Sort left and right parts recursively, excluding known pivot values
1379            doSortByByteKey(a, left,   less - 2, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
1380            doSortByByteKey(a, great + 2, right, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
1381    
1382            /*
1383             * If pivot1 == pivot2, all elements from center
1384             * part are equal and, therefore, already sorted
1385             */
1386            if (!pivotsDiffer) {
1387                return;
1388            }
1389    
1390            /*
1391             * If center part is too large (comprises > 2/3 of the array),
1392             * swap internal pivot values to ends
1393             */
1394            if (less < e1 && great > e5) {
1395                while (a.getByte(less, keyOffset) == pivot1) {
1396                    less++;
1397                }
1398                while (a.getByte(great, keyOffset) == pivot2) {
1399                    great--;
1400                }
1401    
1402                /*
1403                 * Partitioning:
1404                 *
1405                 *   left part       center part                   right part
1406                 * +----------------------------------------------------------+
1407                 * | == pivot1 |  pivot1 < && < pivot2  |    ?    | == pivot2 |
1408                 * +----------------------------------------------------------+
1409                 *              ^                        ^       ^
1410                 *              |                        |       |
1411                 *             less                      k     great
1412                 *
1413                 * Invariants:
1414                 *
1415                 *              all in (*, less)  == pivot1
1416                 *     pivot1 < all in [less, k)   < pivot2
1417                 *              all in (great, *) == pivot2
1418                 *
1419                 * Pointer k is the first index of ?-part
1420                 */
1421                outer:
1422                for (long k = less; k <= great; k++) {
1423                    long ak = a.getByte(k, keyOffset);
1424                    a.get(k, pe1);
1425                    if (ak == pivot2) { // Move a[k] to right part
1426                        while (a.getByte(great, keyOffset) == pivot2) {
1427                            if (great-- == k) {
1428                                break outer;
1429                            }
1430                        }
1431                        if (a.getByte(great, keyOffset) == pivot1) {
1432                            a.get(less, pe3);
1433                            a.set(k, pe3);
1434                            a.get(great, pe3);
1435                            a.set(less++, pe3);
1436                        } else { // pivot1 < a[great] < pivot2
1437                            a.get(great, pe3);
1438                            a.set(k, pe3);
1439                        }
1440                        a.set(great--, pe1);
1441                    } else if (ak == pivot1) { // Move a[k] to left part
1442                        a.get(less, pe3);
1443                        a.set(k, pe3);
1444                        a.set(less++, pe1);
1445                    }
1446                }
1447            }
1448    
1449            // Sort center part recursively, excluding known pivot values
1450            doSortByByteKey(a, less, great, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
1451        }

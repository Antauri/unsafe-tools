017package com.alexkasko.unsafe.offheapstruct;
018
019import java.util.*;
020import java.util.concurrent.Callable;
021import java.util.concurrent.ExecutionException;
022import java.util.concurrent.ExecutorService;
023import java.util.concurrent.Future;
024
025import static com.alexkasko.unsafe.offheapstruct.OffHeapStructSorterWithComparator.doSort;
026
027/**
028 * Naive parallel quicksort implementation on top of {@link OffHeapStructSorterWithComparator}.
029 * Collections is divided in equal parts, each part being sorted separately using user-specified {@link ExecutorService}
030 * and sorted {@link Iterator} is opened over the collection and returned to client.
031 * Note: collection will be left in partially sorted state after iterator opening.
032 */
033class OffHeapStructParallelSorterWithComparator {
034
035    /**
036     * Partially sorts collection and returns fully sorted iterator over it
037     *
038     * @param executor executor service for parallel sorting
039     * @param threads number of worker threads to use
040     * @param a the off-heap struct collection to be sorted
041     * @param comparator structs comparator
042     * @return sorted iterator over the collection
043     * @throws IllegalArgumentException {@code if (fromIndex < 0 || fromIndex > toIndex || toIndex > a.size())}
044     * @throws RuntimeException on worker thread error
045     */
046    static Iterator<byte[]> sortedIterator(ExecutorService executor, int threads, OffHeapStructCollection a, Comparator<OffHeapStructAccessor> comparator) {
047        return sortedIterator(executor, threads, a, 0, a.size(), comparator);
048    }
049
050    /**
051     * Partially sorts part of the collection and returns fully sorted iterator over it
052     *
053     * @param executor executor service for parallel sorting
054     * @param threads number of worker threads to use
055     * @param a the off-heap struct collection to be sorted
056     * @param fromIndex the index of the first element, inclusive, to be sorted
057     * @param toIndex the index of the last element, exclusive, to be sorted
058     * @param comparator structs comparator
059     * @return sorted iterator over the collection part
060     * @throws IllegalArgumentException {@code if (fromIndex < 0 || fromIndex > toIndex || toIndex > a.size())}
061     * @throws RuntimeException on worker thread error
062     */
063    static Iterator<byte[]> sortedIterator(ExecutorService executor, int threads, OffHeapStructCollection a, long fromIndex,
064                                           long toIndex, Comparator<OffHeapStructAccessor> comparator) {
065        if(null == comparator) throw new IllegalArgumentException("Provided comparator is null");
066        if (fromIndex < 0 || fromIndex > toIndex || toIndex > a.size()) {
067            throw new IllegalArgumentException("Illegal input, collection size: [" + a.size() + "], " +
068                    "fromIndex: [" + fromIndex + "], toIndex: [" + toIndex + "]");
069        }
070        OffHeapStructComparator comp = new OffHeapStructComparator(a, comparator);
071        int len = a.structLength();
072        long step = (toIndex - 1 - fromIndex)/threads;
073        if(0 == step) {
074            doSort(a, fromIndex, toIndex - 1, comp, new byte[len], new byte[len], new byte[len], new byte[len], new byte[len],
075                    new byte[len], new byte[len]);
076            return a.iterator();
077        } else {
078            List<Worker> workers = new ArrayList<Worker>();
079            for(int start = 0; start < toIndex; start += step) {
080                Worker worker = new Worker(a, start, Math.min(start + step - 1, toIndex - 1), comp, new byte[len], new byte[len], new byte[len],
081                        new byte[len], new byte[len], new byte[len], new byte[len]);
082                workers.add(worker);
083            }
084            invokeAndWait(executor, workers);
085            return new MergeIter(a, comp, workers);
086        }
087    }
088
089    private static void invokeAndWait(ExecutorService executor, List<Worker> workers) {
090        try {
091            List<Future<Void>> futures = executor.invokeAll(workers);
092            for(Future<Void> fu : futures) fu.get();
093        } catch (InterruptedException e) {
094            throw new RuntimeException(e);
095        } catch (ExecutionException e) {
096            throw new RuntimeException(e);
097        }
098    }
099
100    private static class Worker implements Callable<Void> {
101        private final OffHeapStructCollection a;
102        final long left;
103        final long right;
104        private final OffHeapStructComparator comp;
105        private final byte[] pi;
106        private final byte[] pj;
107        private final byte[] pe1;
108        private final byte[] pe2;
109        private final byte[] pe3;
110        private final byte[] pe4;
111        private final byte[] pe5;
112
113        private Worker(OffHeapStructCollection a, long left, long right, OffHeapStructComparator comp, byte[] pi,
114                       byte[] pj, byte[] pe1, byte[] pe2, byte[] pe3, byte[] pe4, byte[] pe5) {
115            this.a = a;
116            this.left = left;
117            this.right = right;
118            this.comp = comp;
119            this.pi = pi;
120            this.pj = pj;
121            this.pe1 = pe1;
122            this.pe2 = pe2;
123            this.pe3 = pe3;
124            this.pe4 = pe4;
125            this.pe5 = pe5;
126        }
127
128        @Override
129        public Void call() throws Exception {
130            doSort(a, left, right, comp, pi, pj, pe1, pe2, pe3, pe4, pe5);
131            return null;
132        }
133    }
134
135    private static class MergeIter implements Iterator<byte[]> {
136        private final OffHeapStructCollection col;
137        private final OffHeapStructComparator comp;
138        private final List<Worker> bounds;
139        private final long[] indices;
140        private final byte[] buf;
141
142
143        private MergeIter(OffHeapStructCollection col, OffHeapStructComparator comp, List<Worker> bounds) {
144            this.col = col;
145            this.comp = comp;
146            this.bounds = bounds;
147            this.buf = new byte[col.structLength()];
148            this.indices = new long[bounds.size()];
149            for(int i = 0; i < bounds.size(); i++) {
150                indices[i] = bounds.get(i).left;
151            }
152
153        }
154
155        @Override
156        public boolean hasNext() {
157            for (int i = 0; i < indices.length; i++) {
158                if(indices[i] <= bounds.get(i).right) return true;
159            }
160            return false;
161        }
162
163        @Override
164        public byte[] next() {
165            int minind = -1;
166            for (int i = 0; i < indices.length; i++) {
167                if(indices[i] > bounds.get(i).right) continue;
168                if(-1 == minind) minind = i;
169                if(comp.gt(indices[minind], indices[i])) minind = i;
170            }
171            if(-1 == minind) throw new NoSuchElementException();
172            col.get(indices[minind], buf);
173            indices[minind] += 1;
174            return buf;
175        }
176
177        @Override
178        public void remove() {
179            throw new UnsupportedOperationException("remove");
180        }
181    }
182}

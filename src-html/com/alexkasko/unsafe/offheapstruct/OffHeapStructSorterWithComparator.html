017    package com.alexkasko.unsafe.offheapstruct;
018    
019    import java.util.Comparator;
020    
021    import static com.alexkasko.unsafe.offheapstruct.OffHeapStructSorter.INSERTION_SORT_THRESHOLD;
022    
023    /**
024     * <p>alexkasko: borrowed from {@code https://android.googlesource.com/platform/libcore/+/android-4.2.2_r1/luni/src/main/java/java/util/DualPivotQuicksort.java}
025     * and adapted to {@link com.alexkasko.unsafe.offheapstruct.OffHeapStructCollection} with comparator.
026     *
027     * <p>This class implements the Dual-Pivot Quicksort algorithm by
028     * Vladimir Yaroslavskiy, Jon Bentley, and Joshua Bloch. The algorithm
029     * offers O(n log(n)) performance on many data sets that cause other
030     * quicksorts to degrade to quadratic performance, and is typically
031     * faster than traditional (one-pivot) Quicksort implementations.
032     *
033     * @author Vladimir Yaroslavskiy
034     * @author Jon Bentley
035     * @author Josh Bloch
036     *
037     * @version 2009.11.29 m765.827.12i
038     */
039    class OffHeapStructSorterWithComparator {
040    
041        /**
042         * Sorts the specified off-heap struct collection into ascending order using unsigned long struct key.
043         *
044         * @param a the off-heap struct collection to be sorted
045         * @param comparator structs comparator
046         */
047        static void sort(OffHeapStructCollection a, Comparator<OffHeapStructAccessor> comparator) {
048            sort(a, 0, a.size(), comparator);
049        }
050    
051        /**
052         * Sorts the specified range of the off-heap struct collection into ascending order using unsigned long struct key.
053         * The range to be sorted extends from the index {@code fromIndex}, inclusive, to
054         * the index {@code toIndex}, exclusive. If {@code fromIndex == toIndex},
055         * the range to be sorted is empty (and the call is a no-op).
056         *
057         * @param a the off-heap struct collection to be sorted
058         * @param fromIndex the index of the first element, inclusive, to be sorted
059         * @param toIndex the index of the last element, exclusive, to be sorted
060         * @param comparator structs comparator
061         * @throws IllegalArgumentException {@code if (fromIndex < 0 || fromIndex > toIndex || toIndex > a.size())}
062         */
063        static void sort(OffHeapStructCollection a, long fromIndex, long toIndex, Comparator<OffHeapStructAccessor> comparator) {
064            if(null == comparator) throw new IllegalArgumentException("Provided comparator is null");
065            if (fromIndex < 0 || fromIndex > toIndex || toIndex > a.size()) {
066                throw new IllegalArgumentException("Illegal input, collection size: [" + a.size() + "], " +
067                        "fromIndex: [" + fromIndex + "], toIndex: [" + toIndex + "]");
068            }
069            OffHeapStructComparator comp = new OffHeapStructComparator(a, comparator);
070            int len = a.structLength();
071            doSort(a, fromIndex, toIndex - 1, comp, new byte[len], new byte[len], new byte[len], new byte[len], new byte[len],
072                    new byte[len], new byte[len]);
073        }
074    
075        /**
076         * Sorts the specified range of the off-heap struct collection into ascending order using unsigned long struct key.
077         * This method differs from the public {@code sort} method in that the
078         * {@code right} index is inclusive, and it does no range checking on
079         * {@code left} or {@code right}.
080         *
081         * @param a the off-heap header-payload collection to be sorted
082         * @param left the index of the first element, inclusive, to be sorted
083         * @param right the index of the last element, inclusive, to be sorted* @param keyOffset
084         * @param comp structs comparator
085         * @param pi temporary buffer for structs
086         * @param pj temporary buffer for structs
087         * @param pe1 temporary buffer for structs
088         * @param pe2 temporary buffer for structs
089         * @param pe3 temporary buffer for structs
090         * @param pe4 temporary buffer for structs
091         * @param pe5 temporary buffer for structs
092         */
093        private static void doSort(OffHeapStructCollection a, long left, long right, OffHeapStructComparator comp,
094                                   byte[] pi, byte[] pj, byte[] pe1, byte[] pe2, byte[] pe3, byte[] pe4, byte[] pe5) {
095            // Use insertion sort on tiny arrays
096            if (right - left + 1 < INSERTION_SORT_THRESHOLD) {
097                for (long i = left + 1; i <= right; i++) {
098                    a.get(i, pi);
099                    long j;
100                    for (j = i - 1; j >= left && comp.lt(pi, j); j--) {
101                        a.get(j, pj);
102                        a.set(j + 1, pj);
103                    }
104                    a.set(j + 1, pi);
105                }
106            } else { // Use Dual-Pivot Quicksort on large arrays
107                dualPivotQuicksort(a, left, right, comp, pi, pj, pe1, pe2, pe3, pe4, pe5);
108            }
109        }
110    
111        /**
112         * Sorts the specified range of the off-heap struct collection into ascending order by the
113         * Dual-Pivot Quicksort algorithm using unsigned long struct key.
114         *
115         * @param a the off-heap header-payload collection to be sorted
116         * @param left the index of the first element, inclusive, to be sorted
117         * @param right the index of the last element, inclusive, to be sorted
118         * @param comp structs comparator
119         * @param pi temporary buffer for structs
120         * @param pj temporary buffer for structs
121         * @param pe1 temporary buffer for structs
122         * @param pe2 temporary buffer for structs
123         * @param pe3 temporary buffer for structs
124         * @param pe4 temporary buffer for structs
125         * @param pe5 temporary buffer for structs
126         */
127        private static void dualPivotQuicksort(OffHeapStructCollection a, long left, long right, OffHeapStructComparator comp, byte[] pi, byte[] pj,
128                                               byte[] pe1, byte[] pe2, byte[] pe3, byte[] pe4, byte[] pe5) {
129            // Compute indices of five evenly spaced elements
130            long sixth = (right - left + 1) / 6;
131            long e1 = left  + sixth;
132            long e5 = right - sixth;
133            long e3 = (left + right) >>> 1; // The midpoint
134            long e4 = e3 + sixth;
135            long e2 = e3 - sixth;
136    
137            // Sort these elements using a 5-element sorting network
138            long ae1 = e1, ae2 = e2, ae3 = e3, ae4 = e4, ae5 = e5;
139            a.get(e1, pe1); a.get(e2, pe2); a.get(e3, pe3); a.get(e4, pe4); a.get(e5, pe5);
140    
141            if (comp.gt(ae1, ae2)) { long t = ae1; byte[] pt = pe1; ae1 = ae2; pe1 = pe2; ae2 = t; pe2 = pt; }
142            if (comp.gt(ae4, ae5)) { long t = ae4; byte[] pt = pe4; ae4 = ae5; pe4 = pe5; ae5 = t; pe5 = pt; }
143            if (comp.gt(ae1, ae3)) { long t = ae1; byte[] pt = pe1; ae1 = ae3; pe1 = pe3; ae3 = t; pe3 = pt; }
144            if (comp.gt(ae2, ae3)) { long t = ae2; byte[] pt = pe2; ae2 = ae3; pe2 = pe3; ae3 = t; pe3 = pt; }
145            if (comp.gt(ae1, ae4)) { long t = ae1; byte[] pt = pe1; ae1 = ae4; pe1 = pe4; ae4 = t; pe4 = pt; }
146            if (comp.gt(ae3, ae4)) { long t = ae3; byte[] pt = pe3; ae3 = ae4; pe3 = pe4; ae4 = t; pe4 = pt; }
147            if (comp.gt(ae2, ae5)) { long t = ae2; byte[] pt = pe2; ae2 = ae5; pe2 = pe5; ae5 = t; pe5 = pt; }
148            if (comp.gt(ae2, ae3)) { long t = ae2; byte[] pt = pe2; ae2 = ae3; pe2 = pe3; ae3 = t; pe3 = pt; }
149            if (comp.gt(ae4, ae5)) { long t = ae4; byte[] pt = pe4; ae4 = ae5; pe4 = pe5; ae5 = t; pe5 = pt; }
150    
151            a.set(e1, pe1); a.set(e3, pe3); a.set(e5, pe5);
152    
153            /*
154             * Use the second and fourth of the five sorted elements as pivots.
155             * These values are inexpensive approximations of the first and
156             * second terciles of the array. Note that pivot1 <= pivot2.
157             *
158             * The pivots are stored in local variables, and the first and
159             * the last of the elements to be sorted are moved to the locations
160             * formerly occupied by the pivots. When partitioning is complete,
161             * the pivots are swapped back into their final positions, and
162             * excluded from subsequent sorting.
163             */
164            a.get(left, pe1);
165            a.set(e2, pe1);
166            a.get(right, pe1);
167            a.set(e4, pe1);
168    
169            // Pointers
170            long less  = left  + 1; // The index of first element of center part
171            long great = right - 1; // The index before first element of right part
172    
173            // comparable results holders
174            int kpivot1, greatpivot1;
175    
176            boolean pivotsDiffer = !comp.eq(pe2, pe4);
177    
178            if (pivotsDiffer) {
179                /*
180                 * Partitioning:
181                 *
182                 *   left part         center part                    right part
183                 * +------------------------------------------------------------+
184                 * | < pivot1  |  pivot1 <= && <= pivot2  |    ?    |  > pivot2 |
185                 * +------------------------------------------------------------+
186                 *              ^                          ^       ^
187                 *              |                          |       |
188                 *             less                        k     great
189                 *
190                 * Invariants:
191                 *
192                 *              all in (left, less)   < pivot1
193                 *    pivot1 <= all in [less, k)     <= pivot2
194                 *              all in (great, right) > pivot2
195                 *
196                 * Pointer k is the first index of ?-part
197                 */
198                outer:
199                for (long k = less; k <= great; k++) {
200                    a.get(k, pe1);
201                    if (comp.lt(k, pe2)) { // Move a[k] to left part
202                        if (k != less) {
203                            a.get(less, pe3);
204                            a.set(k, pe3);
205                            a.set(less, pe1);
206                        }
207                        less++;
208                    } else if (comp.gt(k, pe4)) { // Move a[k] to right part
209                        while (comp.gt(great, pe4)) {
210                            if (great-- == k) {
211                                break outer;
212                            }
213                        }
214                        if (comp.lt(great, pe2)) {
215                            a.get(less, pe3);
216                            a.set(k, pe3);
217                            a.get(great, pe3);
218                            a.set(less++, pe3);
219                            a.set(great--, pe1);
220                        } else { // pivot1 <= a[great] <= pivot2
221                            a.get(great, pe3);
222                            a.set(k, pe3);
223                            a.set(great--, pe1);
224                        }
225                    }
226                }
227            } else { // Pivots are equal
228                /*
229                 * Partition degenerates to the traditional 3-way,
230                 * or "Dutch National Flag", partition:
231                 *
232                 *   left part   center part            right part
233                 * +----------------------------------------------+
234                 * |  < pivot  |  == pivot  |    ?    |  > pivot  |
235                 * +----------------------------------------------+
236                 *              ^            ^       ^
237                 *              |            |       |
238                 *             less          k     great
239                 *
240                 * Invariants:
241                 *
242                 *   all in (left, less)   < pivot
243                 *   all in [less, k)     == pivot
244                 *   all in (great, right) > pivot
245                 *
246                 * Pointer k is the first index of ?-part
247                 */
248                for (long k = less; k <= great; k++) {
249                    a.get(k, pe1);
250                    kpivot1 = comp.compare(k, pe2);
251                    if (0 == kpivot1) {
252                        continue;
253                    }
254                    if (kpivot1 < 0) { // Move a[k] to left part
255                        if (k != less) {
256                            a.get(less, pe3);
257                            a.set(k, pe3);
258                            a.set(less, pe1);
259                        }
260                        less++;
261                    } else { // (a[k] > pivot1) -  Move a[k] to right part
262                        /*
263                         * We know that pivot1 == a[e3] == pivot2. Thus, we know
264                         * that great will still be >= k when the following loop
265                         * terminates, even though we don't test for it explicitly.
266                         * In other words, a[e3] acts as a sentinel for great.
267                         */
268                        while ((greatpivot1 = comp.compare(great, pe2)) > 0) {
269                            great--;
270                        }
271                        if (greatpivot1 < 0) {
272                            a.get(less, pe3);
273                            a.set(k, pe3);
274                            a.get(great, pe3);
275                            a.set(less++, pe3);
276                            a.set(great--, pe1);
277                        } else { // a[great] == pivot1
278                            a.get(great, pe3);
279                            a.set(k, pe3);
280                            a.set(great--, pe1);
281                        }
282                    }
283                }
284            }
285    
286            // Swap pivots into their final positions
287            a.get(less - 1, pe1);
288            a.set(left, pe1); a.set(less - 1, pe2);
289            a.get(great + 1, pe1);
290            a.set(right, pe1); a.set(great + 1, pe4);
291    
292            // Sort left and right parts recursively, excluding known pivot values
293            doSort(a, left,   less - 2, comp, pi, pj, pe1, pe2, pe3, pe4, pe5);
294            doSort(a, great + 2, right, comp, pi, pj, pe1, pe2, pe3, pe4, pe5);
295    
296            /*
297             * If pivot1 == pivot2, all elements from center
298             * part are equal and, therefore, already sorted
299             */
300            if (!pivotsDiffer) {
301                return;
302            }
303    
304            /*
305             * If center part is too large (comprises > 2/3 of the array),
306             * swap internal pivot values to ends
307             */
308            if (less < e1 && great > e5) {
309                while (comp.eq(less,pe2)) {
310                    less++;
311                }
312                while (comp.eq(great, pe4)) {
313                    great--;
314                }
315    
316                /*
317                 * Partitioning:
318                 *
319                 *   left part       center part                   right part
320                 * +----------------------------------------------------------+
321                 * | == pivot1 |  pivot1 < && < pivot2  |    ?    | == pivot2 |
322                 * +----------------------------------------------------------+
323                 *              ^                        ^       ^
324                 *              |                        |       |
325                 *             less                      k     great
326                 *
327                 * Invariants:
328                 *
329                 *              all in (*, less)  == pivot1
330                 *     pivot1 < all in [less, k)   < pivot2
331                 *              all in (great, *) == pivot2
332                 *
333                 * Pointer k is the first index of ?-part
334                 */
335                outer:
336                for (long k = less; k <= great; k++) {
337                    a.get(k, pe1);
338                    if (comp.eq(k, pe4)) { // Move a[k] to right part
339                        while (comp.eq(great, pe4)) {
340                            if (great-- == k) {
341                                break outer;
342                            }
343                        }
344                        if (comp.eq(great, pe2)) {
345                            a.get(less, pe3);
346                            a.set(k, pe3);
347                            a.get(great, pe3);
348                            a.set(less++, pe3);
349                        } else { // pivot1 < a[great] < pivot2
350                            a.get(great, pe3);
351                            a.set(k, pe3);
352                        }
353                        a.set(great--, pe1);
354                    } else if (comp.eq(k, pe2)) { // Move a[k] to left part
355                        a.get(less, pe3);
356                        a.set(k, pe3);
357                        a.set(less++, pe1);
358                    }
359                }
360            }
361    
362            // Sort center part recursively, excluding known pivot values
363            doSort(a, less, great, comp, pi, pj, pe1, pe2, pe3, pe4, pe5);
364        }

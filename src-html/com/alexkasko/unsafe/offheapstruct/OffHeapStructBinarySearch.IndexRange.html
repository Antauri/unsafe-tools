037         * Performs a binary search for {@code value} in the ascending sorted off-heap struct collection using long struct key.
038         * Searching in an unsorted collection has an undefined result. It's also undefined which element
039         * is found if there are multiple occurrences of the same element.
040         *
041         * @param collection the sorted array to search.
042         * @param value the element to find.
043         * @param keyOffset long key field offset within stuct bounds
044         * @return the non-negative index of the element, or a negative index which
045         *         is {@code -index - 1} where the element would be inserted.
046         */
047        public static long binarySearchByLongKey(OffHeapStructCollection collection, long value, int keyOffset) {
048            return binarySearchByLongKey(collection, 0, collection.size(), value, keyOffset);
049        }

051        /**
052         * Performs a binary search for {@code value} in the ascending sorted off-heap struct collection using long struct key.
053         * in the range specified by fromIndex (inclusive) and toIndex (exclusive).
054         * Searching in an unsorted collection has an undefined result. It's also undefined which element
055         * is found if there are multiple occurrences of the same element.
056         *
057         * @param collection      the sorted collection to search.
058         * @param startIndex the inclusive start index.
059         * @param endIndex   the exclusive end index.
060         * @param value      the element to find.
061         * @param keyOffset  long key field offset within stuct bounds
062         * @return the non-negative index of the element, or a negative index which
063         *         is {@code -index - 1} where the element would be inserted.
064         * @throws IllegalArgumentException {@code if (startIndex < 0 || startIndex > endIndex || endIndex > collection.size()}
065         */
066        public static long binarySearchByLongKey(OffHeapStructCollection collection, long startIndex, long endIndex, long value, int keyOffset) {
067            if (startIndex < 0 || startIndex > endIndex || endIndex > collection.size()) {
068                throw new IllegalArgumentException("Illegal input, collection size: [" + collection.size() + "], " +
069                        "startIndex: [" + startIndex + "], endIndex: [" + endIndex + "]");
070            }
071            long lo = startIndex;
072            long hi = endIndex - 1;
073            while (lo <= hi) {
074                long mid = (lo + hi) >>> 1;
075                long midVal = collection.getLong(mid, keyOffset);
076    
077                if (midVal < value) {
078                    lo = mid + 1;
079                } else if (midVal > value) {
080                    hi = mid - 1;
081                } else {
082                    return mid;  // value found
083                }
084            }
085            return ~lo;  // value not present
086        }

090         * Returns range of indices having given value or empty range.
091         * Searching in an unsorted collection has an undefined result. It's also undefined which element

100            binarySearchRangeByLongKey(collection, 0, collection.size(), value, keyOffset, out);
101        }

103        /**
104         * Performs a binary search for {@code value} in the ascending sorted off-heap struct collection using long struct key.
105         * Returns range of indices having given value or empty range.
106         * Searching in an unsorted collection has an undefined result. It's also undefined which element
107         * is found if there are multiple occurrences of the same element.
108         *
109         * @param collection the sorted array to search.
110         * @param startIndex the inclusive start index.
111         * @param endIndex   the exclusive end index.
112         * @param value the element to find.
113         * @param keyOffset  long key field offset within stuct bounds
114         * @param out range instance, will be set with start/end indices having given value or with empty value
115         */
116        public static void binarySearchRangeByLongKey(OffHeapStructCollection collection, long startIndex, long endIndex,
117                                                   long value, int keyOffset, IndexRange out) {
118            long ind = binarySearchByLongKey(collection, startIndex, endIndex, value, keyOffset);
119            if(ind < 0) {
120                out.setEmpty(ind);
121                return;
122            }
123            long from = ind;
124            while (from >= startIndex && value == collection.getLong(from, keyOffset)) from -= 1;
125            from += 1;
126            long to = ind;
127            while (to < endIndex && value == collection.getLong(to, keyOffset)) to += 1;
128            to -= 1;
129            out.set(from, to);
130        }

433        /**
434         * {@link OffHeapStructCollection} index range representation.
435         * Was made mutable to prevent new object instantiation for each search.
436         * {@link #isEmpty()} method should be checked before accessing indices.
437         * Empty range will contain negative equal indices which values are
438         * {@code -index - 1} where the element would be inserted.
439         * Range with size {@code 1} will contain equal indices.
440         */
441        public static class IndexRange implements Serializable {
442            private static final long serialVersionUID = 879348143026038919L;
443            private boolean empty;
444            private long fromIndex;
445            private long toIndex;
446    
447            public IndexRange() {
448                this.empty = false;
449                this.fromIndex = -1;
450                this.toIndex = -1;
451            }
452    
453            /**
454             * Sets value for empty range
455             *
456             * @param value negative value for empty range returned by search
457             */
458            private void setEmpty(long value) {
459                this.empty = true;
460                this.fromIndex = value;
461                this.toIndex = value;
462            }
463    
464            /**
465             * Sets boundaries for non-empty range
466             *
467             * @param from start index
468             * @param to end index
469             */
470            private void set(long from, long to) {
471                this.empty = false;
472                this.fromIndex = from;
473                this.toIndex = to;
474            }
475    
476            /**
477             * Whether this range is empty
478             *
479             * @return whether this range is empty
480             */
481            public boolean isEmpty() {
482                return empty;
483            }
484    
485            /**
486             * Whether this range is not empty
487             *
488             * @return whether this range is not empty
489             */
490            public boolean isNotEmpty() {
491                return !empty;
492            }
493    
494            /**
495             * Start index or {@code -index - 1}
496             *      where the element would be inserted for empty range
497             *
498             * @return start index value or {@code -index - 1}
499             *      where the element would be inserted for empty range
500             */
501            public long getFromIndex() {
502                return fromIndex;
503            }
504    
505            /**
506             * End index or {@code -index - 1}
507             *      where the element would be inserted for empty range
508             *
509             * @return end index value, {@code -index - 1}
510             *      where the element would be inserted for empty range
511             */
512            public long getToIndex() {
513                return toIndex;
514            }
515    
516            /**
517             * {@inheritDoc}
518             */
519            @Override
520            public boolean equals(Object o) {
521                if (this == o) return true;
522                if (o == null || getClass() != o.getClass()) return false;
523                IndexRange that = (IndexRange) o;
524                if (empty != that.empty) return false;
525                if (fromIndex != that.fromIndex) return false;
526                if (toIndex != that.toIndex) return false;
527                return true;
528            }
529    
530            /**
531             * {@inheritDoc}
532             */
533            @Override
534            public int hashCode() {
535                int result = (empty ? 1 : 0);
536                result = 31 * result + (int) (fromIndex ^ (fromIndex >>> 32));
537                result = 31 * result + (int) (toIndex ^ (toIndex >>> 32));
538                return result;
539            }
540    
541            /**
542             * {@inheritDoc}
543             */
544            @Override
545            public String toString() {
546                final StringBuilder sb = new StringBuilder();
547                sb.append("IndexRange");
548                sb.append("{empty=").append(empty);
549                sb.append(", fromIndex=").append(fromIndex);
550                sb.append(", toIndex=").append(toIndex);
551                sb.append('}');
552                return sb.toString();
553            }
554        }

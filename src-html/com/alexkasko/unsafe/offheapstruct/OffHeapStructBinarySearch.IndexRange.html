034        /**
035         * Performs a binary search for {@code value} in the ascending sorted off-heap struct collection using int struct key.
036         * Searching in an unsorted collection has an undefined result. It's also undefined which element
037         * is found if there are multiple occurrences of the same element.
038         *
039         * @param collection the sorted array to search.
040         * @param value the element to find.
041         * @param keyOffset int key field offset within stuct bounds
042         * @return the non-negative index of the element, or a negative index which
043         *         is {@code -index - 1} where the element would be inserted.
044         */
045        public static long binarySearchByIntKey(OffHeapStructCollection collection, long value, int keyOffset) {
046            return binarySearchByIntKey(collection, 0, collection.size(), value, keyOffset);
047        }
048    
049        /**
050         * Performs a binary search for {@code value} in the ascending sorted off-heap struct collection using int struct key.
051         * in the range specified by fromIndex (inclusive) and toIndex (exclusive).
052         * Searching in an unsorted collection has an undefined result. It's also undefined which element
053         * is found if there are multiple occurrences of the same element.
054         *
055         * @param collection      the sorted collection to search.
056         * @param startIndex the inclusive start index.
057         * @param endIndex   the exclusive end index.
058         * @param value      the element to find.
059         * @param keyOffset  int key field offset within stuct bounds
060         * @return the non-negative index of the element, or a negative index which
061         *         is {@code -index - 1} where the element would be inserted.
062         * @throws IllegalArgumentException {@code if (startIndex < 0 || startIndex > endIndex || endIndex > collection.size()}
063         */
064        public static long binarySearchByIntKey(OffHeapStructCollection collection, long startIndex, long endIndex, long value, int keyOffset) {
065            if (startIndex < 0 || startIndex > endIndex || endIndex > collection.size()) {
066                throw new IllegalArgumentException("Illegal input, collection size: [" + collection.size() + "], " +
067                        "startIndex: [" + startIndex + "], endIndex: [" + endIndex + "]");
068            }
069            long lo = startIndex;
070            long hi = endIndex - 1;
071            while (lo <= hi) {
072                long mid = (lo + hi) >>> 1;
073                long midVal = collection.getInt(mid, keyOffset);
074    
075                if (midVal < value) {
076                    lo = mid + 1;
077                } else if (midVal > value) {
078                    hi = mid - 1;
079                } else {
080                    return mid;  // value found
081                }
082            }
083            return ~lo;  // value not present
084        }

086        /**
087         * Performs a binary search for {@code value} in the ascending sorted off-heap struct collection using int struct key.
088         * Returns range of indices having given value or empty range.
089         * Searching in an unsorted collection has an undefined result. It's also undefined which element
090         * is found if there are multiple occurrences of the same element.
091         *
092         * @param collection the sorted array to search.
093         * @param value the element to find.
094         * @param keyOffset  int key field offset within stuct bounds
095         * @param out range instance, will be set with start/end indices having given value or with empty value
096         */
097        public static void binarySearchRangeByIntKey(OffHeapStructCollection collection, long value, int keyOffset, IndexRange out) {
098            binarySearchRangeByIntKey(collection, 0, collection.size(), value, keyOffset, out);
099        }
100    
101        /**
102         * Performs a binary search for {@code value} in the ascending sorted off-heap struct collection using int struct key.
103         * Returns range of indices having given value or empty range.
104         * Searching in an unsorted collection has an undefined result. It's also undefined which element
105         * is found if there are multiple occurrences of the same element.
106         *
107         * @param collection the sorted array to search.
108         * @param startIndex the inclusive start index.
109         * @param endIndex   the exclusive end index.
110         * @param value the element to find.
111         * @param keyOffset  int key field offset within stuct bounds
112         * @param out range instance, will be set with start/end indices having given value or with empty value
113         */
114        public static void binarySearchRangeByIntKey(OffHeapStructCollection collection, long startIndex, long endIndex,
115                                                   long value, int keyOffset, IndexRange out) {
116            long ind = binarySearchByIntKey(collection, startIndex, endIndex, value, keyOffset);
117            if(ind < 0) {
118                out.setEmpty(ind);
119                return;
120            }
121            long from = ind;
122            while (from >= startIndex && value == collection.getInt(from, keyOffset)) from -= 1;
123            from += 1;
124            long to = ind;
125            while (to < endIndex && value == collection.getInt(to, keyOffset)) to += 1;
126            to -= 1;
127            out.set(from, to);
128        }

228        /**
229         * {@link OffHeapStructCollection} index range representation.
230         * Was made mutable to prevent new object instantiation for each search.
231         * {@link #isEmpty()} method should be checked before accessing indices.
232         * Empty range will contain negative equal indices which values are
233         * {@code -index - 1} where the element would be inserted.
234         * Range with size {@code 1} will contain equal indices.
235         */
236        public static class IndexRange implements Serializable {
237            private static final long serialVersionUID = 879348143026038919L;
238            private boolean empty;
239            private long fromIndex;
240            private long toIndex;
241    
242            public IndexRange() {
243                this.empty = false;
244                this.fromIndex = -1;
245                this.toIndex = -1;
246            }
247    
248            /**
249             * Sets value for empty range
250             *
251             * @param value negative value for empty range returned by search
252             */
253            private void setEmpty(long value) {
254                this.empty = true;
255                this.fromIndex = value;
256                this.toIndex = value;
257            }
258    
259            /**
260             * Sets boundaries for non-empty range
261             *
262             * @param from start index
263             * @param to end index
264             */
265            private void set(long from, long to) {
266                this.empty = false;
267                this.fromIndex = from;
268                this.toIndex = to;
269            }
270    
271            /**
272             * Whether this range is empty
273             *
274             * @return whether this range is empty
275             */
276            public boolean isEmpty() {
277                return empty;
278            }
279    
280            /**
281             * Whether this range is not empty
282             *
283             * @return whether this range is not empty
284             */
285            public boolean isNotEmpty() {
286                return !empty;
287            }
288    
289            /**
290             * Start index or {@code -index - 1}
291             *      where the element would be inserted for empty range
292             *
293             * @return start index value or {@code -index - 1}
294             *      where the element would be inserted for empty range
295             */
296            public long getFromIndex() {
297                return fromIndex;
298            }
299    
300            /**
301             * End index or {@code -index - 1}
302             *      where the element would be inserted for empty range
303             *
304             * @return end index value, {@code -index - 1}
305             *      where the element would be inserted for empty range
306             */
307            public long getToIndex() {
308                return toIndex;
309            }
310    
311            /**
312             * {@inheritDoc}
313             */
314            @Override
315            public boolean equals(Object o) {
316                if (this == o) return true;
317                if (o == null || getClass() != o.getClass()) return false;
318                IndexRange that = (IndexRange) o;
319                if (empty != that.empty) return false;
320                if (fromIndex != that.fromIndex) return false;
321                if (toIndex != that.toIndex) return false;
322                return true;
323            }
324    
325            /**
326             * {@inheritDoc}
327             */
328            @Override
329            public int hashCode() {
330                int result = (empty ? 1 : 0);
331                result = 31 * result + (int) (fromIndex ^ (fromIndex >>> 32));
332                result = 31 * result + (int) (toIndex ^ (toIndex >>> 32));
333                return result;
334            }
335    
336            /**
337             * {@inheritDoc}
338             */
339            @Override
340            public String toString() {
341                final StringBuilder sb = new StringBuilder();
342                sb.append("IndexRange");
343                sb.append("{empty=").append(empty);
344                sb.append(", fromIndex=").append(fromIndex);
345                sb.append(", toIndex=").append(toIndex);
346                sb.append('}');
347                return sb.toString();
348            }
349        }

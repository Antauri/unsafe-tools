001    package com.alexkasko.unsafe.offheap;
002    
003    /**
004     * Implementation of off-heap header-payload array. Memory block for each index
005     * contains {@code long} header anb {@code byte[]} payload.
006     *
007     * Default implementation uses {@code sun.misc.Unsafe}, with all operations guarded with {@code assert} keyword.
008     * With assertions enabled in runtime ({@code -ea} java switch) {@link AssertionError}
009     * will be thrown on illegal index access. Without assertions illegal index will crash JVM.
010     *
011     * Array won't be zeroed after creation (will contain garbage by default).
012     * Allocated memory may be freed manually using {@link #free()} (thread-safe
013     * and may be called multiple times) or it will be freed after {@link OffHeapLongArray}
014     * will be garbage collected.
015     *
016     * @author alexkasko
017     * Date: 3/3/13
018     */
019    public class OffHeapPayloadArray implements OffHeapPayloadAddressable, OffHeapDisposable {
020        private static final int HEADER_LENGTH = 8;
021    
022        private final OffHeapMemory ohm;
023        private final int payloadLength;
024        private final int elementLength;
025    
026        /**
027         * Constructor
028         *
029         * @param size array size
030         * @param payloadLength length of payload data block in bytes
031         */
032        public OffHeapPayloadArray(long size, int payloadLength) {
033            if(payloadLength <= 0) throw new IllegalArgumentException("Illegal payloadLength: [" + payloadLength + "]");
034            this.payloadLength = payloadLength;
035            this.elementLength = HEADER_LENGTH + payloadLength;
036            this.ohm = OffHeapMemory.allocateMemory(size * elementLength);
037        }
038    
039        /**
040         * Whether unsafe implementation of {@link OffHeapMemory} is used
041         *
042         * @return whether unsafe implementation of {@link OffHeapMemory} is used
043         */
044        public boolean isUnsafe() {
045            return ohm.isUnsafe();
046        }
047    
048        /**
049         * Returns length of the each payload block in bytes
050         *
051         * @return length of the each payload block in bytes
052         */
053        @Override
054        public int payloadLength() {
055            return payloadLength;
056        }
057    
058        /**
059         * Gets the header at position {@code index}
060         *
061         * @param index collection index
062         * @return long value
063         */
064        @Override
065        public long get(long index) {
066            return ohm.getLong(index * elementLength);
067        }
068    
069        /**
070         * Loads payload into provided buffer
071         *
072         * @param index collection index to load payload from
073         * @param buffer buffer to load payload into
074         */
075        @Override
076        public void getPayload(long index, byte[] buffer) {
077            ohm.get(index * elementLength + HEADER_LENGTH, buffer);
078        }
079    
080        /**
081         * Loads payload into provided buffer
082         *
083         * @param index collection index to load payload from
084         * @param buffer buffer to load payload into
085         * @param bufferPos buffer offset
086         */
087        public void getPayload(long index, byte[] buffer, int bufferPos) {
088            ohm.get(index * elementLength + HEADER_LENGTH, buffer, bufferPos, payloadLength);
089        }
090    
091        /**
092         * Sets header value and copies payload data on provided index
093         *
094         * @param index collection index to set header and payload
095         * @param header header value
096         * @param payload payload value
097         */
098        @Override
099        public void set(long index, long header, byte[] payload) {
100            long addr = index * elementLength;
101            ohm.putLong(addr, header);
102            ohm.put(addr + HEADER_LENGTH, payload);
103        }
104    
105        /**
106         * Sets header value and copies payload data on provided index
107         *
108         * @param index collection index to set header and payload
109         * @param header header value
110         * @param payload payload value
111         * @param payloadPos payload offset
112         */
113        public void set(long index, long header, byte[] payload, int payloadPos) {
114            long addr = index * elementLength;
115            ohm.putLong(addr, header);
116            ohm.put(addr + HEADER_LENGTH, payload, payloadPos, payloadLength);
117        }
118    
119        /**
120         * Returns number of elements in array
121         *
122         * @return number of elements in array
123         */
124        @Override
125        public long size() {
126            return ohm.length() / elementLength;
127        }
128    
129        /**
130         * Frees allocated memory, may be called multiple times from any thread
131         */
132        @Override
133        public void free() {
134            ohm.free();
135        }
136    
137        /**
138         * {@inheritDoc}
139         */
140        @Override
141        public String toString() {
142            final StringBuilder sb = new StringBuilder();
143            sb.append("OffHeapPayloadArray");
144            sb.append("{size=").append(size());
145            sb.append(", unsafe=").append(isUnsafe());
146            sb.append(", payloadLength=").append(payloadLength);
147            sb.append('}');
148            return sb.toString();
149        }
150    }

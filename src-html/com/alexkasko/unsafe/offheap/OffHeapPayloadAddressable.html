001    package com.alexkasko.unsafe.offheap;
002    
003    /**
004     * Interface for off-heap header-payload collection.
005     *
006     * @author alexkasko
007     * Date: 3/4/13
008     */
009    public interface OffHeapPayloadAddressable extends OffHeapAddressable {
010    
011        /**
012         * Returns length of the each payload block in bytes
013         *
014         * @return length of the each payload block in bytes
015         */
016        int payloadLength();
017    
018        /**
019         * Loads payload into provided buffer
020         *
021         * @param index collection index to load payload from
022         * @param buffer buffer to load payload into
023         */
024        void getPayload(long index, byte[] buffer);
025    
026        /**
027         * Sets header value and copies payload data on provided index
028         *
029         * @param index collection index to set header and payload
030         * @param header header value
031         * @param payload payload value
032         */
033        void set(long index, long header, byte[] payload);
034    }

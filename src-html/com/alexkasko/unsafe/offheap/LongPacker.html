001    package com.alexkasko.unsafe.offheap;
002    
003    /**
004     * Contains static methods for packing two values with limited size (one long and one int)
005     * into one long value. May be used to pack values before placing them into {@link OffHeapLongAddressable}.
006     * Values are stored using bit shifting as follows:
007     * <pre>
008     * {@code
009     *  0                             32        64-bits             64
010     *  +-----------------------------+-----------+------------------+
011     *  | first 32 bits of long value | int value | tail of long val |
012     *  +-----------------------------+-----------+------------------+
013     * }
014     * </pre>
015     * This class doesn't use {@code sun.misc.Unsafe}
016     *
017     * @author alexkasko
018     * Date: 1/14/13
019     */
020    public class LongPacker {
021        /**
022         * Packs one long value with limited size and one int value
023         * into one long value.
024         * Values limits are checking using {@code assert}
025         * keyword, java must be run with {@code -ea} switch to enable them.
026         *
027         * @param big long value
028         * @param little int value
029         * @param bits max size of long value in bits, max size of int value is {@code 64 - bits}
030         * @return packed long value
031         */
032        public static long pack(long big, int little, int bits) {
033            assert bits > 32 && bits < 64;
034            assert big < (1L << bits);
035            assert little < (1 << (64 - bits));
036            int ls = bits % 8;
037            int bm = (1 << ls) - 1;
038            long res = (big & ~bm) << (64 - bits);
039            res |= (little & ((1L << (64 - bits)) - 1)) << ls;
040            res |= big & bm;
041            return res;
042        }
043    
044        /**
045         * Read long value from packed long.
046         * Values limits are checking using {@code assert}
047         * keyword, java must be run with {@code -ea} switch to enable them.
048         *
049         * @param pack packed long
050         * @param bits max size of long value in bits, max size of int value is {@code 64 - bits}
051         * @return source long value
052         */
053        public static long big(long pack, int bits) {
054            assert bits > 32 && bits < 64;
055            int ls = bits % 8;
056            long res = (pack & (((1L << (bits - ls)) - 1) << (64 - bits + ls))) >>> (64 - bits);
057            return res | pack & ((1 << ls) - 1);
058        }
059    
060        /**
061         *
062         * Values limits are checking using {@code assert}
063         * keyword, java must be run with {@code -ea} switch to enable them.
064         *
065         * @param pack packed long
066         * @param bits max size of long value in bits, max size of int value is {@code 64 - bits}
067         * @return source int value
068         */
069        public static int little(long pack, int bits) {
070            assert bits > 32 && bits < 64;
071            int ls = bits % 8;
072            return (int) ((pack & (((1L << (64 - bits)) -1) << ls)) >> ls);
073        }
074    }

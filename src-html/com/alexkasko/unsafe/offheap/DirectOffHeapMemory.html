017    package com.alexkasko.unsafe.offheap;
018    
019    import java.lang.reflect.Method;
020    import java.nio.ByteBuffer;
021    import java.util.concurrent.atomic.AtomicBoolean;
022    
023    import static java.nio.ByteOrder.LITTLE_ENDIAN;
024    
025    /**
026     * Implementation of {@link OffHeapMemory} using {@link ByteBuffer#allocateDirect(int)}
027     *
028     * @author alexkasko
029     * Date: 1/14/13
030    */
031    class DirectOffHeapMemory extends OffHeapMemory {
032    
033        private final ByteBuffer bb;
034        private final long length;
035        private Object cleaner;
036        private Method clean;
037        private final AtomicBoolean disposed = new AtomicBoolean(false);
038    
039        DirectOffHeapMemory(long bytes) {
040            this.length = bytes;
041            this.bb = ByteBuffer.allocateDirect((int) bytes).order(LITTLE_ENDIAN);
042            // http://stackoverflow.com/a/8191493/314015
043            try {
044                setupOpenJdkCleaner();
045            } catch (Exception e) {
046                try {
047                    setupAndroidCleaner();
048                } catch (Exception e1) {
049                    e.printStackTrace();
050                    e1.printStackTrace();
051                    throw new RuntimeException(e);
052                }
053            }
054        }
055    
056        private void setupOpenJdkCleaner() throws Exception {
057            Method cleanerMethod = bb.getClass().getMethod("cleaner");
058            cleanerMethod.setAccessible(true);
059            this.cleaner = cleanerMethod.invoke(bb);
060            this.clean = cleaner.getClass().getMethod("clean");
061            this.clean.setAccessible(true);
062        }
063    
064        private void setupAndroidCleaner() throws Exception {
065            this.clean = bb.getClass().getMethod("free");
066            this.clean.setAccessible(true);
067            this.cleaner = bb;
068        }
069    
070        /**
071         * {@inheritDoc}
072         */
073        @Override
074        public boolean isUnsafe() {
075            return false;
076        }
077    
078        /**
079         * {@inheritDoc}
080         */
081        @Override
082        public long length() {
083            return length;
084        }
085    
086        /**
087         * {@inheritDoc}
088         */
089        @Override
090        public void free() {
091            if(!disposed.compareAndSet(false, true)) return;
092            try {
093                clean.invoke(cleaner);
094            } catch (Throwable e) {
095                e.printStackTrace();
096            }
097        }
098    
099        /**
100         * {@inheritDoc}
101         */
102        @Override
103        public void put(long offset, byte[] buffer, int bufferOffset, int bytes) {
104            bb.clear().position((int) offset);
105            bb.put(buffer, bufferOffset, bytes);
106        }
107    
108        /**
109         * {@inheritDoc}
110         */
111        @Override
112        public void put(long offset, byte[] buffer) {
113            bb.clear().position((int) offset);
114            bb.put(buffer);
115        }
116    
117        /**
118         * {@inheritDoc}
119         */
120        @Override
121        public void get(long offset, byte[] buffer, int bufferOffset, int bytes) {
122            bb.clear().position((int) offset);
123            bb.get(buffer, bufferOffset, bytes);
124        }
125    
126        /**
127         * {@inheritDoc}
128         */
129        @Override
130        public void get(long offset, byte[] buffer) {
131            bb.clear().position((int) offset);
132            bb.get(buffer);
133        }
134    
135        /**
136         * {@inheritDoc}
137         */
138        @Override
139        public byte getByte(long offset) {
140            return bb.get((int) offset);
141        }
142    
143        /**
144         * {@inheritDoc}
145         */
146        @Override
147        public void putByte(long offset, byte value) {
148            bb.put((int) offset, value);
149        }
150    
151        /**
152         * {@inheritDoc}
153         */
154        @Override
155        public short getUnsignedByte(long offset) {
156            return (short) (bb.get((int) offset) & 0xff);
157        }
158    
159        /**
160         * {@inheritDoc}
161         */
162        @Override
163        public void putUnsignedByte(long offset, short value) {
164            bb.put((int) offset, (byte) value);
165        }
166    
167        /**
168         * {@inheritDoc}
169         */
170        @Override
171        public short getShort(long offset) {
172            return bb.getShort((int) offset);
173        }
174    
175        /**
176         * {@inheritDoc}
177         */
178        @Override
179        public void putShort(long offset, short value) {
180            bb.putShort((int) offset, value);
181        }
182    
183        /**
184         * {@inheritDoc}
185         */
186        @Override
187        public int getUnsignedShort(long offset) {
188            return bb.getShort((int) offset) & 0xffff;
189        }
190    
191        /**
192         * {@inheritDoc}
193         */
194        @Override
195        public void putUnsignedShort(long offset, int value) {
196            bb.putShort((int) offset, (short) value);
197        }
198    
199        /**
200         * {@inheritDoc}
201         */
202        @Override
203        public int getInt(long offset) {
204            return bb.getInt((int) offset);
205        }
206    
207        /**
208         * {@inheritDoc}
209         */
210        @Override
211        public void putInt(long offset, int value) {
212            bb.putInt((int) offset, value);
213        }
214    
215        /**
216         * {@inheritDoc}
217         */
218        @Override
219        public long getUnsignedInt(long offset) {
220            return bb.getInt((int) offset) & 0xffffffffL;
221        }
222    
223        /**
224         * {@inheritDoc}
225         */
226        @Override
227        public void putUnsignedInt(long offset, long value) {
228            bb.putInt((int) offset, (int) value);
229        }
230    
231        /**
232         * {@inheritDoc}
233         */
234        @Override
235        public long getLong(long offset) {
236            return bb.getLong((int) offset);
237        }
238    
239        /**
240         * {@inheritDoc}
241         */
242        @Override
243        public void putLong(long offset, long value) {
244            bb.putLong((int) offset, value);
245        }
246    
247        /**
248         * {@inheritDoc}
249         */
250        @Override
251        public void copy(long offset, OffHeapMemory destination, long destOffset, long bytes) {
252            DirectOffHeapMemory dest = (DirectOffHeapMemory) destination;
253            bb.clear().position((int) offset);
254            dest.bb.clear().position((int) destOffset);
255            dest.bb.put(bb);
256        }
257    
258        /**
259         * {@inheritDoc}
260         */
261        @Override
262        public String toString() {
263            final StringBuilder sb = new StringBuilder();
264            sb.append("DirectOffHeapMemory");
265            sb.append("{bb=").append(bb);
266            sb.append(", length=").append(length);
267            sb.append(", cleaner=").append(cleaner);
268            sb.append(", clean=").append(clean);
269            sb.append(", disposed=").append(disposed);
270            sb.append('}');
271            return sb.toString();
272        }
273    }

001    package com.alexkasko.unsafe.offheap;
002    
003    /**
004     * Basic interface for off-heap long-indexed collections. Required for {@link OffHeapBinarySearch}.
005     *
006     * @author alexkasko
007     * Date: 3/5/13
008     */
009    public interface OffHeapAddressable {
010        /**
011         * Gets the element at position {@code index}
012         *
013         * @param index collection index
014         * @return long value
015         */
016        long get(long index);
017    
018        /**
019         * Returns number of elements in collection
020         *
021         * @return number of elements in collection
022         */
023        long size();
024    }

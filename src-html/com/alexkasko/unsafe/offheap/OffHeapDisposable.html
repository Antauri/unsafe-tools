001    package com.alexkasko.unsafe.offheap;
002    
003    /**
004     * Basic interface for off-heap classes that support manual memory freeing.
005     * Not freed memory will be released on {@link OffHeapMemory} (impl) instances garbage collection.
006     *
007     * @author alexkasko
008     * Date: 3/5/13
009     */
010    public interface OffHeapDisposable {
011        /**
012         * Frees allocated memory, may be called multiple times from any thread
013         */
014        void free();
015    }

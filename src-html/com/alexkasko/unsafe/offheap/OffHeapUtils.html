017    package com.alexkasko.unsafe.offheap;
018    
019    /**
020     * Static utility methods for off-heap classes
021     *
022     * @author alexkasko
023     * Date: 3/5/13
024     */
025    public class OffHeapUtils {
026        /**
027         * Utility method for usage in {@code finally} clauses.
028         * Calls {@link com.alexkasko.unsafe.offheap.OffHeapDisposable#free()}
029         * on provided instance only if instance is not null.
030         *
031         * @param disposable off-heap disposable instance, may be null
032         */
033        public static void free(OffHeapDisposable disposable) {
034            if(null == disposable) return;
035            disposable.free();
036        }
037    }

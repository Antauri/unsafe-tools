001    package com.alexkasko.unsafe.offheap;
002    
003    /**
004     * Static utility methods for off-heap classes
005     *
006     * @author alexkasko
007     * Date: 3/5/13
008     */
009    public class OffHeapUtils {
010        /**
011         * Utility method for usage in {@code finally} clauses.
012         * Calls {@link com.alexkasko.unsafe.offheap.OffHeapDisposable#free()}
013         * on provided instance only if instance is not null.
014         *
015         * @param disposable off-heap disposable instance, may be null
016         */
017        public static void free(OffHeapDisposable disposable) {
018            if(null == disposable) return;
019            disposable.free();
020        }
021    }

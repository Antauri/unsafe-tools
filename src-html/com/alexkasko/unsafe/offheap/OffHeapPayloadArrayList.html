001    package com.alexkasko.unsafe.offheap;
002    
003    /**
004     * Implementation of off-heap header-payload array-list of long using {@link OffHeapMemory}.
005     * Memory block for each index contains {@code long} header anb {@code byte[]} payload.
006     * Memory area will be allocated another time and copied on elements adding. This class doesn't support elements removing.
007     * {@link #get(long)} and {@link #set(long, long, byte[])} access operations indexes are checked using {@code assert} keyword
008     * (indexes between size and capacity will be rejected).
009     *
010     * Default implementation uses {@code sun.misc.Unsafe}, with all operations guarded with {@code assert} keyword.
011     * With assertions enabled in runtime ({@code -ea} java switch) {@link AssertionError}
012     * will be thrown on illegal index access. Without assertions illegal index will crash JVM.
013     *
014     * Array won't be zeroed after creation (will contain garbage by default).
015     * Allocated memory may be freed manually using {@link #free()} (thread-safe
016     * and may be called multiple times) or it will be freed after {@link OffHeapLongArray}
017     * will be garbage collected.
018     *
019     * @author alexkasko
020     * Date: 3/4/13
021     */
022    public class OffHeapPayloadArrayList implements OffHeapPayloadAddressable, OffHeapDisposable {
023        private static final int MIN_CAPACITY_INCREMENT = 12;
024        private static final int HEADER_LENGTH = 8;
025    
026        private final int payloadLength;
027        private final int elementLength;
028        private OffHeapMemory ohm;
029        private long size;
030    
031        /**
032         * Constructor, {@code 12} is used as initial capacity
033         *
034         * @param payloadLength length of payload data block in bytes
035         */
036        public OffHeapPayloadArrayList(int payloadLength) {
037            this(MIN_CAPACITY_INCREMENT, payloadLength);
038        }
039    
040        /**
041         * Constructor
042         *
043         * @param capacity initial capacity
044         * @param payloadLength length of payload data block in bytes
045         */
046        public OffHeapPayloadArrayList(long capacity, int payloadLength) {
047            if(payloadLength <= 0) throw new IllegalArgumentException("Illegal payloadLength: [" + payloadLength + "]");
048            this.payloadLength = payloadLength;
049            this.elementLength = HEADER_LENGTH + payloadLength;
050            this.ohm = OffHeapMemory.allocateMemory(capacity * elementLength);
051        }
052    
053        /**
054         * Whether unsafe implementation of {@link OffHeapMemory} is used
055         *
056         * @return whether unsafe implementation of {@link OffHeapMemory} is used
057         */
058        public boolean isUnsafe() {
059            return ohm.isUnsafe();
060        }
061    
062        public void add(long header, byte[] payload) {
063            add(header, payload, 0);
064        }
065    
066        /**
067         * Adds element to the end of this list. Memory area will be allocated another time and copied
068         * on capacity exceed.
069         *
070         * @param header header to add
071         * @param payload payload to add
072         * @param payloadPos payload offset
073         */
074        public void add(long header, byte[] payload, int payloadPos) {
075            OffHeapMemory oh = ohm;
076            long s = size;
077            if (s == capacity()) {
078                long len = s + (s < (MIN_CAPACITY_INCREMENT / 2) ? MIN_CAPACITY_INCREMENT : s >> 1);
079                OffHeapMemory newOhm = OffHeapMemory.allocateMemory(len * elementLength);
080                // maybe it's better to use Unsafe#reallocateMemory here
081                oh.copy(0, newOhm, 0, ohm.length());
082                oh.free();
083                ohm = newOhm;
084            }
085            size = s + 1;
086            set(s, header, payload, payloadPos);
087        }
088    
089        /**
090         * Returns length of the each payload block in bytes
091         *
092         * @return length of the each payload block in bytes
093         */
094        @Override
095        public int payloadLength() {
096            return payloadLength;
097        }
098    
099        /**
100         * Gets the header at position {@code index}
101         *
102         * @param index list index
103         * @return long header
104         */
105        @Override
106        public long get(long index) {
107            assert index < size : index;
108            return ohm.getLong(index * elementLength);
109        }
110    
111        /**
112         * Loads payload into provided buffer
113         *
114         * @param index list index to load payload from
115         * @param buffer buffer to load payload into
116         */
117        @Override
118        public void getPayload(long index, byte[] buffer) {
119            assert index < size : index;
120            ohm.get(index * elementLength + HEADER_LENGTH, buffer);
121        }
122    
123        /**
124         * Loads payload into provided buffer
125         *
126         * @param index list index to load payload from
127         * @param buffer buffer to load payload into
128         * @param bufferPos buffer position
129         */
130        public void getPayload(long index, byte[] buffer, int bufferPos) {
131            assert index < size : index;
132            ohm.get(index * elementLength + HEADER_LENGTH, buffer, bufferPos, payloadLength);
133        }
134    
135        /**
136         * Sets header value and copies payload data on provided index
137         *
138         * @param index list index to set header and payload
139         * @param header header value
140         * @param payload payload value
141         */
142        @Override
143        public void set(long index, long header, byte[] payload) {
144            assert index < size : index;
145            long addr = index * elementLength;
146            ohm.putLong(addr, header);
147            ohm.put(addr + HEADER_LENGTH, payload);
148        }
149    
150        /**
151         * Sets header value and copies payload data on provided index
152         *
153         * @param index list index to set header and payload
154         * @param header header value
155         * @param payload payload value
156         * @param payloadPos payload position
157         */
158        public void set(long index, long header, byte[] payload, int payloadPos) {
159            assert index < size : index;
160            long addr = index * elementLength;
161            ohm.putLong(addr, header);
162            ohm.put(addr + HEADER_LENGTH, payload, payloadPos, payloadLength);
163        }
164    
165        /**
166         * Copies payload data on provided index
167         *
168         * @param index collection index to set payload
169         * @param payload payload value
170         */
171        public void setPayload(long index, byte[] payload) {
172            assert index < size : index;
173            long addr = index * elementLength;
174            ohm.put(addr + HEADER_LENGTH, payload);
175        }
176    
177        /**
178         * Copies payload data on provided index
179         *
180         * @param index collection index to set payload
181         * @param payload payload value
182         * @param payloadPos payload offset
183         */
184        public void setPayload(long index, byte[] payload, int payloadPos) {
185            assert index < size : index;
186            long addr = index * elementLength;
187            ohm.put(addr + HEADER_LENGTH, payload, payloadPos, payloadLength);
188        }
189    
190    
191        /**
192         * Returns number of elements in list
193         *
194         * @return number of elements in list
195         */
196        @Override
197        public long size() {
198            return size;
199        }
200    
201        /**
202         * Returns number of elements list may contain without additional memory allocation
203         *
204         * @return number of elements list may contain without additional memory allocation
205         */
206        public long capacity() {
207            return ohm.length() / elementLength;
208        }
209    
210        /**
211         * Frees allocated memory, may be called multiple times from any thread
212         */
213        @Override
214        public void free() {
215            ohm.free();
216        }
217    
218        /**
219         * {@inheritDoc}
220         */
221        @Override
222        public String toString() {
223            final StringBuilder sb = new StringBuilder();
224            sb.append("OffHeapPayloadArrayList");
225            sb.append("{size=").append(size);
226            sb.append(", unsafe=").append(isUnsafe());
227            sb.append(", payloadLength=").append(payloadLength);
228            sb.append('}');
229            return sb.toString();
230        }
231    }

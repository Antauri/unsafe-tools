001    package com.alexkasko.unsafe.offheap;
002    
003    /**
004     * Implementation of off-heap header-int_payload array. Memory block for each index
005     * contains {@code long} header anb {@code int} payload.
006     *
007     * Default implementation uses {@code sun.misc.Unsafe}, with all operations guarded with {@code assert} keyword.
008     * With assertions enabled in runtime ({@code -ea} java switch) {@link AssertionError}
009     * will be thrown on illegal index access. Without assertions illegal index will crash JVM.
010     *
011     * Array won't be zeroed after creation (will contain garbage by default).
012     * Allocated memory may be freed manually using {@link #free()} (thread-safe
013     * and may be called multiple times) or it will be freed after {@link com.alexkasko.unsafe.offheap.OffHeapLongArray}
014     * will be garbage collected.
015     *
016     * @author alexkasko
017     * Date: 4/24/13
018     */
019    public class OffHeapPayloadIntArray implements OffHeapPayloadIntAddressable, OffHeapDisposable {
020        private static final int HEADER_LENGTH = 8;
021        private static final int ELEMENT_LENGTH = 12;
022    
023        private final OffHeapMemory ohm;
024    
025        /**
026         * Constructor
027         *
028         * @param size array size
029         */
030        public OffHeapPayloadIntArray(long size) {
031            this.ohm = OffHeapMemory.allocateMemory(size * ELEMENT_LENGTH);
032        }
033    
034        /**
035         * Whether unsafe implementation of {@link com.alexkasko.unsafe.offheap.OffHeapMemory} is used
036         *
037         * @return whether unsafe implementation of {@link com.alexkasko.unsafe.offheap.OffHeapMemory} is used
038         */
039        public boolean isUnsafe() {
040            return ohm.isUnsafe();
041        }
042    
043        /**
044         * Gets the header at position {@code index}
045         *
046         * @param index collection index
047         * @return long value
048         */
049        @Override
050        public long get(long index) {
051            return ohm.getLong(index * ELEMENT_LENGTH);
052        }
053    
054        /**
055         * Returns payload for the given index
056         *
057         * @param index collection index to load payload from
058         */
059        @Override
060        public int getPayload(long index) {
061            return ohm.getInt(index * ELEMENT_LENGTH + HEADER_LENGTH);
062        }
063    
064        /**
065         * Sets header and payload values on specified index
066         *
067         * @param index collection index to set header and payload
068         * @param header header value
069         * @param payload payload value
070         */
071        @Override
072        public void set(long index, long header, int payload) {
073            long addr = index * ELEMENT_LENGTH;
074            ohm.putLong(addr, header);
075            ohm.putInt(addr + HEADER_LENGTH, payload);
076        }
077    
078        /**
079         * Sets payload value on specified index
080         *
081         * @param index collection index to set payload
082         * @param payload payload value
083         */
084        public void setPayload(long index, int payload) {
085            long addr = index * ELEMENT_LENGTH;
086            ohm.putInt(addr + HEADER_LENGTH, payload);
087        }
088    
089        /**
090         * Returns number of elements in array
091         *
092         * @return number of elements in array
093         */
094        @Override
095        public long size() {
096            return ohm.length() / ELEMENT_LENGTH;
097        }
098    
099        /**
100         * Frees allocated memory, may be called multiple times from any thread
101         */
102        @Override
103        public void free() {
104            ohm.free();
105        }
106    
107        /**
108         * {@inheritDoc}
109         */
110        @Override
111        public String toString() {
112            final StringBuilder sb = new StringBuilder();
113            sb.append("OffHeapPayloadIntArray");
114            sb.append("{size=").append(size());
115            sb.append(", unsafe=").append(isUnsafe());
116            sb.append('}');
117            return sb.toString();
118        }
119    }

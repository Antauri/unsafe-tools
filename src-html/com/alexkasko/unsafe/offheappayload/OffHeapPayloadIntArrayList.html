017package com.alexkasko.unsafe.offheappayload;
018
019import com.alexkasko.unsafe.offheap.OffHeapDisposable;
020import com.alexkasko.unsafe.offheap.OffHeapMemory;
021
022/**
023 * Implementation of off-heap header-int_payload array-list of long using {@link com.alexkasko.unsafe.offheap.OffHeapMemory}.
024 * Memory block for each index contains {@code long} header anb {@code int} payload.
025 * Memory area will be allocated another time and copied on elements adding. This class doesn't support elements removing.
026 * {@link #get(long)} and {@link #set(long, long, int)} access operations indexes are checked using {@code assert} keyword
027 * (indexes between size and capacity will be rejected).
028 *
029 * Default implementation uses {@code sun.misc.Unsafe}, with all operations guarded with {@code assert} keyword.
030 * With assertions enabled in runtime ({@code -ea} java switch) {@link AssertionError}
031 * will be thrown on illegal index access. Without assertions illegal index will crash JVM.
032 *
033 * Array won't be zeroed after creation (will contain garbage by default).
034 * Allocated memory may be freed manually using {@link #free()} (thread-safe
035 * and may be called multiple times) or it will be freed after {@link com.alexkasko.unsafe.offheaplong.OffHeapLongArray}
036 * will be garbage collected.
037 *
038 * @author alexkasko
039 * Date: 4/24/13
040 */
041@Deprecated // use offheapstruct package
042public class OffHeapPayloadIntArrayList implements OffHeapPayloadIntAddressable, OffHeapDisposable {
043    private static final int MIN_CAPACITY_INCREMENT = 12;
044    private static final int HEADER_LENGTH = 8;
045    private static final int ELEMENT_LENGTH = 12;
046
047    private OffHeapMemory ohm;
048    private long size;
049
050    /**
051     * Constructor, {@code 12} is used as initial capacity
052     */
053    public OffHeapPayloadIntArrayList() {
054        this(MIN_CAPACITY_INCREMENT);
055    }
056
057    /**
058     * Constructor
059     *
060     * @param capacity initial capacity
061     */
062    public OffHeapPayloadIntArrayList(long capacity) {
063        this.ohm = OffHeapMemory.allocateMemory(capacity * ELEMENT_LENGTH);
064    }
065
066    /**
067     * Whether unsafe implementation of {@link com.alexkasko.unsafe.offheap.OffHeapMemory} is used
068     *
069     * @return whether unsafe implementation of {@link com.alexkasko.unsafe.offheap.OffHeapMemory} is used
070     */
071    public boolean isUnsafe() {
072        return ohm.isUnsafe();
073    }
074
075    /**
076     * Adds element to the end of this list. Memory area will be allocated another time and copied
077     * on capacity exceed.
078     *
079     * @param header header to add
080     * @param payload payload to add
081     */
082    public void add(long header, int payload) {
083        OffHeapMemory oh = ohm;
084        long s = size;
085        if (s == capacity()) {
086            long len = s + (s < (MIN_CAPACITY_INCREMENT / 2) ? MIN_CAPACITY_INCREMENT : s >> 1);
087            OffHeapMemory newOhm = OffHeapMemory.allocateMemory(len * ELEMENT_LENGTH);
088            // maybe it's better to use Unsafe#reallocateMemory here
089            oh.copy(0, newOhm, 0, ohm.length());
090            oh.free();
091            ohm = newOhm;
092        }
093        size = s + 1;
094        set(s, header, payload);
095    }
096
097    /**
098     * Gets the header at position {@code index}
099     *
100     * @param index list index
101     * @return long header
102     */
103    @Override
104    public long get(long index) {
105        assert index < size : index;
106        return ohm.getLong(index * ELEMENT_LENGTH);
107    }
108
109    /**
110     * Returns payload for the given index
111     *
112     * @param index list index to load payload from
113     */
114    @Override
115    public int getPayload(long index) {
116        assert index < size : index;
117        return ohm.getInt(index * ELEMENT_LENGTH + HEADER_LENGTH);
118    }
119
120    /**
121     * Sets header and payload values on specified index
122     *
123     * @param index list index to set header and payload
124     * @param header header value
125     * @param payload payload value
126     */
127    @Override
128    public void set(long index, long header, int payload) {
129        assert index < size : index;
130        long addr = index * ELEMENT_LENGTH;
131        ohm.putLong(addr, header);
132        ohm.putInt(addr + HEADER_LENGTH, payload);
133    }
134
135    /**
136     * Sets payload value on specified index
137     *
138     * @param index collection index to set payload
139     * @param payload payload value
140     */
141    public void setPayload(long index, int payload) {
142        assert index < size : index;
143        long addr = index * ELEMENT_LENGTH;
144        ohm.putInt(addr + HEADER_LENGTH, payload);
145    }
146
147    /**
148     * Returns number of elements in list
149     *
150     * @return number of elements in list
151     */
152    @Override
153    public long size() {
154        return size;
155    }
156
157    /**
158     * Returns number of elements list may contain without additional memory allocation
159     *
160     * @return number of elements list may contain without additional memory allocation
161     */
162    public long capacity() {
163        return ohm.length() / ELEMENT_LENGTH;
164    }
165
166    /**
167     * Frees allocated memory, may be called multiple times from any thread
168     */
169    @Override
170    public void free() {
171        ohm.free();
172    }
173
174    /**
175     * {@inheritDoc}
176     */
177    @Override
178    public String toString() {
179        final StringBuilder sb = new StringBuilder();
180        sb.append("OffHeapPayloadIntArrayList");
181        sb.append("{size=").append(size);
182        sb.append(", unsafe=").append(isUnsafe());
183        sb.append('}');
184        return sb.toString();
185    }
186}

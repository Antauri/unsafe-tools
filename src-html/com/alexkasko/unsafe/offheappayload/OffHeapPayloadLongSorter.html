017    package com.alexkasko.unsafe.offheappayload;
018    
019    /**
020    * alexkasko: borrowed from {@code https://android.googlesource.com/platform/libcore/+/android-4.2.2_r1/luni/src/main/java/java/util/DualPivotQuicksort.java}
021    * and adapted to {@link OffHeapPayloadLongAddressable}.
022    *
023    * This class implements the Dual-Pivot Quicksort algorithm by
024    * Vladimir Yaroslavskiy, Jon Bentley, and Joshua Bloch. The algorithm
025    * offers O(n log(n)) performance on many data sets that cause other
026    * quicksorts to degrade to quadratic performance, and is typically
027    * faster than traditional (one-pivot) Quicksort implementations.
028    *
029    * @author Vladimir Yaroslavskiy
030    * @author Jon Bentley
031    * @author Josh Bloch
032    *
033    * @version 2009.11.29 m765.827.12i
034    */
035    @Deprecated // use offheapstruct package
036    public class OffHeapPayloadLongSorter {
037    
038        /**
039         * If the length of an collection to be sorted is less than this
040         * constant, insertion sort is used in preference to Quicksort.
041         */
042        private static final int INSERTION_SORT_THRESHOLD = 32;
043    
044        /**
045         * Sorts the specified off-heap header-long_payload collection into ascending order.
046         *
047         * @param a the off-heap header-long_payload collection to be sorted
048         */
049        public static void sort(OffHeapPayloadLongAddressable a) {
050            sort(a, 0, a.size());
051        }
052    
053        /**
054         * Sorts the specified range of the off-heap header-long_payload collection into ascending order. The range
055         * to be sorted extends from the index {@code fromIndex}, inclusive, to
056         * the index {@code toIndex}, exclusive. If {@code fromIndex == toIndex},
057         * the range to be sorted is empty (and the call is a no-op).
058         *
059         * @param a the off-heap header-long_payload collection to be sorted
060         * @param fromIndex the index of the first element, inclusive, to be sorted
061         * @param toIndex the index of the last element, exclusive, to be sorted
062         * @throws IllegalArgumentException {@code if (fromIndex < 0 || fromIndex > toIndex || toIndex > a.size())}
063         */
064        public static void sort(OffHeapPayloadLongAddressable a, long fromIndex, long toIndex) {
065            if (fromIndex < 0 || fromIndex > toIndex || toIndex > a.size()) {
066                throw new IllegalArgumentException("Illegal input, collection size: [" + a.size() + "], " +
067                        "fromIndex: [" + fromIndex + "], toIndex: [" + toIndex + "]");
068            }
069            doSort(a, fromIndex, toIndex - 1);
070        }
071    
072        /**
073         * Sorts the specified range of the off-heap header-long_payload collection into ascending order. This
074         * method differs from the public {@code sort} method in that the
075         * {@code right} index is inclusive, and it does no range checking on
076         * {@code left} or {@code right}.
077         *
078         * @param a the off-heap header-long_payload collection to be sorted
079         * @param left the index of the first element, inclusive, to be sorted
080         * @param right the index of the last element, inclusive, to be sorted
081         */
082        private static void doSort(OffHeapPayloadLongAddressable a, long left, long right) {
083    
084            // Use insertion sort on tiny arrays
085            if (right - left + 1 < INSERTION_SORT_THRESHOLD) {
086                for (long i = left + 1; i <= right; i++) {
087                    long ai = a.get(i);
088                    long pi = a.getPayload(i);
089                    long j;
090                    for (j = i - 1; j >= left && ai < a.get(j); j--) {
091                        a.set(j + 1, a.get(j), a.getPayload(j));
092                    }
093                    a.set(j + 1, ai, pi);
094                }
095            } else { // Use Dual-Pivot Quicksort on large arrays
096                dualPivotQuicksort(a, left, right);
097            }
098        }
099    
100        /**
101         * Sorts the specified range of the off-heap header-long_payload collection into ascending order by the
102         * Dual-Pivot Quicksort algorithm.
103         *
104         * @param a the off-heap header-long_payload collection to be sorted
105         * @param left the index of the first element, inclusive, to be sorted
106         * @param right the index of the last element, inclusive, to be sorted
107         */
108        private static void dualPivotQuicksort(OffHeapPayloadLongAddressable a, long left, long right) {
109            // Compute indices of five evenly spaced elements
110            long sixth = (right - left + 1) / 6;
111            long e1 = left  + sixth;
112            long e5 = right - sixth;
113            long e3 = (left + right) >>> 1; // The midpoint
114            long e4 = e3 + sixth;
115            long e2 = e3 - sixth;
116    
117            // Sort these elements using a 5-element sorting network
118            long ae1 = a.get(e1), ae2 = a.get(e2), ae3 = a.get(e3), ae4 = a.get(e4), ae5 = a.get(e5);
119            long pe1 = a.getPayload(e1); long pe2 = a.getPayload(e2); long pe3 = a.getPayload(e3);
120            long pe4 = a.getPayload(e4); long pe5 = a.getPayload(e5);
121    
122            if (ae1 > ae2) { long t = ae1; long pt = pe1; ae1 = ae2; pe1 = pe2; ae2 = t; pe2 = pt; }
123            if (ae4 > ae5) { long t = ae4; long pt = pe4; ae4 = ae5; pe4 = pe5; ae5 = t; pe5 = pt; }
124            if (ae1 > ae3) { long t = ae1; long pt = pe1; ae1 = ae3; pe1 = pe3; ae3 = t; pe3 = pt; }
125            if (ae2 > ae3) { long t = ae2; long pt = pe2; ae2 = ae3; pe2 = pe3; ae3 = t; pe3 = pt; }
126            if (ae1 > ae4) { long t = ae1; long pt = pe1; ae1 = ae4; pe1 = pe4; ae4 = t; pe4 = pt; }
127            if (ae3 > ae4) { long t = ae3; long pt = pe3; ae3 = ae4; pe3 = pe4; ae4 = t; pe4 = pt; }
128            if (ae2 > ae5) { long t = ae2; long pt = pe2; ae2 = ae5; pe2 = pe5; ae5 = t; pe5 = pt; }
129            if (ae2 > ae3) { long t = ae2; long pt = pe2; ae2 = ae3; pe2 = pe3; ae3 = t; pe3 = pt; }
130            if (ae4 > ae5) { long t = ae4; long pt = pe4; ae4 = ae5; pe4 = pe5; ae5 = t; pe5 = pt; }
131    
132            a.set(e1, ae1, pe1); a.set(e3, ae3, pe3); a.set(e5, ae5, pe5);
133    
134            /*
135             * Use the second and fourth of the five sorted elements as pivots.
136             * These values are inexpensive approximations of the first and
137             * second terciles of the array. Note that pivot1 <= pivot2.
138             *
139             * The pivots are stored in local variables, and the first and
140             * the last of the elements to be sorted are moved to the locations
141             * formerly occupied by the pivots. When partitioning is complete,
142             * the pivots are swapped back into their final positions, and
143             * excluded from subsequent sorting.
144             */
145            long pivot1 = ae2; a.set(e2, a.get(left), a.getPayload(left));
146            long pivot2 = ae4; a.set(e4, a.get(right), a.getPayload(right));
147    
148            // Pointers
149            long less  = left  + 1; // The index of first element of center part
150            long great = right - 1; // The index before first element of right part
151    
152            boolean pivotsDiffer = (pivot1 != pivot2);
153    
154            if (pivotsDiffer) {
155                /*
156                 * Partitioning:
157                 *
158                 *   left part         center part                    right part
159                 * +------------------------------------------------------------+
160                 * | < pivot1  |  pivot1 <= && <= pivot2  |    ?    |  > pivot2 |
161                 * +------------------------------------------------------------+
162                 *              ^                          ^       ^
163                 *              |                          |       |
164                 *             less                        k     great
165                 *
166                 * Invariants:
167                 *
168                 *              all in (left, less)   < pivot1
169                 *    pivot1 <= all in [less, k)     <= pivot2
170                 *              all in (great, right) > pivot2
171                 *
172                 * Pointer k is the first index of ?-part
173                 */
174                outer:
175                for (long k = less; k <= great; k++) {
176                    long ak = a.get(k);
177                    long pk = a.getPayload(k);
178                    if (ak < pivot1) { // Move a[k] to left part
179                        if (k != less) {
180                            a.set(k, a.get(less), a.getPayload(less));
181                            a.set(less, ak, pk);
182                        }
183                        less++;
184                    } else if (ak > pivot2) { // Move a[k] to right part
185                        while (a.get(great) > pivot2) {
186                            if (great-- == k) {
187                                break outer;
188                            }
189                        }
190                        if (a.get(great) < pivot1) {
191                            a.set(k, a.get(less), a.getPayload(less));
192                            a.set(less++, a.get(great), a.getPayload(great));
193                            a.set(great--, ak, pk);
194                        } else { // pivot1 <= a[great] <= pivot2
195                            a.set(k, a.get(great), a.getPayload(great));
196                            a.set(great--, ak, pk);
197                        }
198                    }
199                }
200            } else { // Pivots are equal
201                /*
202                 * Partition degenerates to the traditional 3-way,
203                 * or "Dutch National Flag", partition:
204                 *
205                 *   left part   center part            right part
206                 * +----------------------------------------------+
207                 * |  < pivot  |  == pivot  |    ?    |  > pivot  |
208                 * +----------------------------------------------+
209                 *              ^            ^       ^
210                 *              |            |       |
211                 *             less          k     great
212                 *
213                 * Invariants:
214                 *
215                 *   all in (left, less)   < pivot
216                 *   all in [less, k)     == pivot
217                 *   all in (great, right) > pivot
218                 *
219                 * Pointer k is the first index of ?-part
220                 */
221                for (long k = less; k <= great; k++) {
222                    long ak = a.get(k);
223                    long pk = a.getPayload(k);
224                    if (ak == pivot1) {
225                        continue;
226                    }
227                    if (ak < pivot1) { // Move a[k] to left part
228                        if (k != less) {
229                            a.set(k, a.get(less), a.getPayload(less));
230                            a.set(less, ak, pk);
231                        }
232                        less++;
233                    } else { // (a[k] > pivot1) -  Move a[k] to right part
234                        /*
235                         * We know that pivot1 == a[e3] == pivot2. Thus, we know
236                         * that great will still be >= k when the following loop
237                         * terminates, even though we don't test for it explicitly.
238                         * In other words, a[e3] acts as a sentinel for great.
239                         */
240                        while (a.get(great) > pivot1) {
241                            great--;
242                        }
243                        if (a.get(great) < pivot1) {
244                            a.set(k, a.get(less), a.getPayload(less));
245                            a.set(less++, a.get(great), a.getPayload(great));
246                            a.set(great--, ak, pk);
247                        } else { // a[great] == pivot1
248                            a.set(k, pivot1, a.getPayload(great));
249                            a.set(great--, ak, pk);
250                        }
251                    }
252                }
253            }
254    
255            // Swap pivots into their final positions
256            a.set(left, a.get(less - 1), a.getPayload(less - 1)); a.set(less - 1, pivot1, pe2);
257            a.set(right, a.get(great + 1), a.getPayload(great + 1)); a.set(great + 1, pivot2, pe4);
258    
259            // Sort left and right parts recursively, excluding known pivot values
260            doSort(a, left,   less - 2);
261            doSort(a, great + 2, right);
262    
263            /*
264             * If pivot1 == pivot2, all elements from center
265             * part are equal and, therefore, already sorted
266             */
267            if (!pivotsDiffer) {
268                return;
269            }
270    
271            /*
272             * If center part is too large (comprises > 2/3 of the array),
273             * swap internal pivot values to ends
274             */
275            if (less < e1 && great > e5) {
276                while (a.get(less) == pivot1) {
277                    less++;
278                }
279                while (a.get(great) == pivot2) {
280                    great--;
281                }
282    
283                /*
284                 * Partitioning:
285                 *
286                 *   left part       center part                   right part
287                 * +----------------------------------------------------------+
288                 * | == pivot1 |  pivot1 < && < pivot2  |    ?    | == pivot2 |
289                 * +----------------------------------------------------------+
290                 *              ^                        ^       ^
291                 *              |                        |       |
292                 *             less                      k     great
293                 *
294                 * Invariants:
295                 *
296                 *              all in (*, less)  == pivot1
297                 *     pivot1 < all in [less, k)   < pivot2
298                 *              all in (great, *) == pivot2
299                 *
300                 * Pointer k is the first index of ?-part
301                 */
302                outer:
303                for (long k = less; k <= great; k++) {
304                    long ak = a.get(k);
305                    long pk = a.getPayload(k);
306                    if (ak == pivot2) { // Move a[k] to right part
307                        while (a.get(great) == pivot2) {
308                            if (great-- == k) {
309                                break outer;
310                            }
311                        }
312                        if (a.get(great) == pivot1) {
313                            a.set(k, a.get(less), a.getPayload(less));
314                            a.set(less++, pivot1, a.getPayload(great));
315                        } else { // pivot1 < a[great] < pivot2
316                            a.set(k, a.get(great), a.getPayload(great));
317                        }
318                        a.set(great--, pivot2, pk);
319                    } else if (ak == pivot1) { // Move a[k] to left part
320                        a.set(k, a.get(less), a.getPayload(less));
321                        a.set(less++, pivot1, pk);
322                    }
323                }
324            }
325    
326            // Sort center part recursively, excluding known pivot values
327            doSort(a, less, great);
328        }

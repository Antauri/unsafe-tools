017    package com.alexkasko.unsafe.offheaplong;
018    
019    import com.alexkasko.unsafe.offheap.OffHeapDisposable;
020    import com.alexkasko.unsafe.offheap.OffHeapMemory;
021    
022    import java.util.Iterator;
023    
024    /**
025     * <p>Implementation of array of long using {@link com.alexkasko.unsafe.offheap.OffHeapMemory}.
026     *
027     * <p>Default implementation uses {@code sun.misc.Unsafe}, with all operations guarded with {@code assert} keyword.
028     * With assertions enabled in runtime ({@code -ea} java switch) {@link AssertionError}
029     * will be thrown on illegal index access. Without assertions illegal index will crash JVM.
030     *
031     * <p>Array won't be zeroed after creation (will contain garbage by default).
032     * Allocated memory may be freed manually using {@link #free()} (thread-safe
033     * and may be called multiple times) or it will be freed after {@link OffHeapLongArray}
034     * will be garbage collected.
035     *
036     * <p>Note: while class implements Iterable, iterator will create new autoboxed Long object
037     * <b>on every</b> {@code next()} call, this behaviour is inevitable with iterators in java 6/7.
038     *
039     * @author alexkasko
040     * Date: 2/22/13
041     */
042    public class OffHeapLongArray implements OffHeapLongAddressable, OffHeapDisposable, Iterable<Long> {
043        private static final int ELEMENT_LENGTH = 8;
044    
045        private final OffHeapMemory ohm;
046    
047        /**
048         * Constructor
049         *
050         * @param size number of elements in array
051         */
052        public OffHeapLongArray(long size) {
053            this.ohm = OffHeapMemory.allocateMemory(size * ELEMENT_LENGTH);
054        }
055    
056        /**
057         * Whether unsafe implementation of {@link OffHeapMemory} is used
058         *
059         * @return whether unsafe implementation of {@link OffHeapMemory} is used
060         */
061        public boolean isUnsafe() {
062            return ohm.isUnsafe();
063        }
064    
065        /**
066         * Gets the element at position {@code index}
067         *
068         * @param index array index
069         * @return long value
070         */
071        @Override
072        public long get(long index) {
073            return ohm.getLong(index * ELEMENT_LENGTH);
074        }
075    
076        /**
077         * Sets the element at position {@code index} to the given value
078         *
079         * @param index array index
080         * @param value long value
081         */
082        @Override
083        public void set(long index, long value) {
084            ohm.putLong(index * ELEMENT_LENGTH, value);
085        }
086    
087        /**
088         * Returns number of elements in array
089         *
090         * @return number of elements in array
091         */
092        @Override
093        public long size() {
094            return ohm.length() / ELEMENT_LENGTH;
095        }
096    
097        /**
098         * Frees allocated memory, may be called multiple times from any thread
099         */
100        public void free() {
101            ohm.free();
102        }
103    
104        /**
105         * {@inheritDoc}
106         */
107        @Override
108        public Iterator<Long> iterator() {
109            return new OffHeapLongIterator(this);
110        }
111    
112        /**
113         * {@inheritDoc}
114         */
115        @Override
116        public String toString() {
117            final StringBuilder sb = new StringBuilder();
118            sb.append("OffHeapLongArray");
119            sb.append("{size=").append(size());
120            sb.append(", unsafe=").append(isUnsafe());
121            sb.append('}');
122            return sb.toString();
123        }
124    }
